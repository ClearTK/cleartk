050    public abstract class LIBSVMClassifier<INPUTOUTCOME_TYPE,OUTPUTOUTCOME_TYPE> extends Classifier_ImplBase<INPUTOUTCOME_TYPE,OUTPUTOUTCOME_TYPE,FeatureVector> {
051            
052            public static final String MODEL_NAME = "model.libsvm";
053            public static final String ATTRIBUTES_NAME = "LIBSVM";
054            public static final String SCALE_FEATURES_KEY = "scaleFeatures";
055            public static final String SCALE_FEATURES_VALUE_NORMALIZEL2 = "normalizeL2";
056            
057            public LIBSVMClassifier(JarFile modelFile) throws IOException {
058                    super(modelFile);
059                    
060                    ZipEntry modelEntry = modelFile.getEntry(LIBSVMClassifier.MODEL_NAME);
061                    File tmpFile = File.createTempFile("tmp", ".mdl");
062                    try {
063                            LIBSVMClassifier.copy(modelFile.getInputStream(modelEntry), new FileOutputStream(tmpFile));
064                            this.model = libsvm.svm.svm_load_model(tmpFile.getPath());
065                    } finally {
066                            tmpFile.delete();
067                    }
068            }
069    
070            public INPUTOUTCOME_TYPE classify(List<Feature> features) throws CleartkException {
071                    FeatureVector featureVector = this.featuresEncoder.encodeAll(features);
072    
073                    OUTPUTOUTCOME_TYPE encodedOutcome = decodePrediction(libsvm.svm.svm_predict(this.model, convertToLIBSVM(featureVector)));
074                    
075                    return outcomeEncoder.decode(encodedOutcome);
076            }
077            
078            
079            private static void copy(InputStream in, OutputStream out) throws IOException {
080                    byte[] buffer = new byte[4096];
081                    int len = 0;
082                    
083                    while( (len = in.read(buffer)) > 0 ) {
084                            out.write(buffer, 0, len);
085                    }
086            }
087            
088            protected static libsvm.svm_node[] convertToLIBSVM(FeatureVector featureVector) {
089                    List<libsvm.svm_node> nodes = new ArrayList<libsvm.svm_node>();
090                    
091                    for( FeatureVector.Entry entry : featureVector ) {
092                            libsvm.svm_node node = new libsvm.svm_node();
093                            node.index = entry.index;
094                            node.value = entry.value;
095                            nodes.add(node);
096                    }
097                    
098                    return nodes.toArray(new libsvm.svm_node[nodes.size()]);
099            }
100            
101            protected abstract OUTPUTOUTCOME_TYPE decodePrediction(double prediction);
102    
103            protected libsvm.svm_model model;
104    }
