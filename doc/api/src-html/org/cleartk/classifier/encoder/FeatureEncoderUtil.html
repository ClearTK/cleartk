033    
034    public class FeatureEncoderUtil {
035    
036            public static String escape(String source) {
037                    return escape(source, new char[0]);
038            }
039    
040            public static String escape(String source, char[] escapeCharacters) {
041                            StringBuffer escapedBuffer = new StringBuffer();
042                            StringCharacterIterator it = new StringCharacterIterator(source);
043                            
044                            for( char c = it.first(); c != StringCharacterIterator.DONE; c = it.next() ) {
045                                    /*
046                                     * 92 is \
047                                     * 37 is %
048                                     * 33 is A
049                                     * 126 is ~
050                                     */
051                                    boolean escaped = false;
052                                    if( c < 33 || c > 126 || c == 92 || c == 37) {
053                                            escapedBuffer.append(String.format("%%U%04X", (int) c));
054                                            escaped = true;
055                                    }
056                                    if(!escaped) {
057                                            for(char esc : escapeCharacters) {
058                                                    if(c == esc) {
059                                                            escapedBuffer.append(String.format("%%U%04X", (int) c));
060                                                            escaped = true;
061                                                            break;
062                                                    }
063                                            }
064                                    }
065                                    if(!escaped)
066                                            escapedBuffer.append(c);
067                            }
068    
069                            return escapedBuffer.toString();
070    
071            }
