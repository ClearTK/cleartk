026    import org.apache.uima.UimaContext;

028    import org.apache.uima.analysis_engine.AnalysisEngineProcessException;

030    import org.apache.uima.resource.ResourceInitializationException;
031    import org.cleartk.util.UIMAUtil;
032

038     * @author Steven Bethard
039     */
040    public abstract class InstanceConsumer_ImplBase<OUTCOME_TYPE> extends JCasAnnotator_ImplBase implements InstanceConsumer<OUTCOME_TYPE> {
041    
042            /**
043             * The name of a AnnotationHandler which will generate classification
044             * instances for each document.
045             */
046            public static final String PARAM_ANNOTATION_HANDLER = "AnnotationHandler";
047    
048            protected AnnotationHandler<OUTCOME_TYPE> annotationHandler;
049    
050            @Override
051            public void initialize(UimaContext context) throws ResourceInitializationException {
052                    super.initialize(context);
053    
054                    // get the class name from the parameter
055                    Object className = UIMAUtil.getRequiredConfigParameterValue(context,
056                                    InstanceConsumer_ImplBase.PARAM_ANNOTATION_HANDLER);
057    
058                    // create a new instance of the AnnotationHandler
059                    try {
060                            Class<?> cls = Class.forName((String) className);
061                            this.annotationHandler = this.getProducerClass(cls).newInstance();
062                    } catch (Exception e) {
063                            throw new ResourceInitializationException(e);
064                    }
065                    
066                    // TODO: do a runtime type check of AnnotationHandler<OUTCOME_TYPE> vs. InstanceConsumer<OUTCOME_TYPE>
067    
068                    // initialize the AnnotationHandler
069                    this.annotationHandler.initialize(context);
070            }
071    
072            @Override
073            public void process(JCas jCas) throws AnalysisEngineProcessException {
074                    this.annotationHandler.process(jCas, this);
075            }
076    
077            @SuppressWarnings("unchecked")
078            private Class<? extends AnnotationHandler<OUTCOME_TYPE>> getProducerClass(Class<?> cls) {
079                    return (Class<? extends AnnotationHandler<OUTCOME_TYPE>>) cls.asSubclass(AnnotationHandler.class);
080            }
081    }
