024    package org.cleartk.classifier;
025    
026    import java.io.IOException;
027    import java.lang.reflect.InvocationTargetException;
028    import java.util.jar.JarFile;
029    
030    /**
031     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado 
032     * <br>All rights reserved.
033    
034    */
035    
036    public class ClassifierFactory {
037    
038            /**
039             * Every classifier should be able to instantiated from a jar file. This
040             * factory method reads the class of the classifier from manifest of the jar
041             * file from the attribute "classifier". e.g.:
042             * <p>
043             * <code>classifier: org.cleartk.classifier.OpenNLPMaxentClassifier</code>
044             * </p>
045             * The value of the classifier attribute of the manifest is used to
046             * instantiate a classifier using reflection and the constructor that takes
047             * a jar file.
048             * 
049             * @param jarFileName
050             *            the name of a jar file
051             * @return a classifier defined by the contents of the jar file.
052             * @throws IOException
053             */
054            public static Classifier<?> readFromJar(String jarFileName) throws IOException {
055                    // get the jar file manifest
056                    JarFile modelFile = new JarFile(jarFileName);
057                    ClassifierManifest manifest = new ClassifierManifest(modelFile);
058            
059                    // get the classifier class
060                    ClassifierBuilder<?> builder = manifest.getClassifierBuilder();
061                    Class<? extends Classifier<?>> classifierClass = builder.getClassifierClass();
062                    
063                    // create the classifier, passing in the jar file
064                    try {
065                            return classifierClass.getConstructor(JarFile.class).newInstance(modelFile);
066                    } catch (InstantiationException e) {
067                            throw newIOException(e);
068                    } catch (IllegalAccessException e) {
069                            throw newIOException(e);
070                    } catch (InvocationTargetException e) {
071                            throw newIOException(e);
072                    } catch (NoSuchMethodException e) {
073                            throw newIOException(e);
074                    } finally {
075                            modelFile.close();
076                    }
077            }
078            
079            private static IOException newIOException(Throwable cause) {
080                    IOException exception = new IOException();
081                    exception.initCause(cause);
082                    return exception;
083            }
084    
085    }
