025    
026    import java.util.Collections;
027    import java.util.List;
028    import java.util.regex.Pattern;
029    
030    import org.cleartk.classifier.Feature;
031    
032    /**

041    public class NumericTypeProliferator extends FeatureProliferator{
042    
043            public static final String DEFAULT_NAME = "NumericType";
044    
045            /**
046             * all characters are digits.
047             */
048            public static final String DIGITS = "DIGITS";
049            /**
050             * all characters are digits and it looks like a year - i.e. string is
051             * 4 digits starting with 1 or 20 or 21.
052             */
053            public static final String YEAR_DIGITS = "YEAR_DIGITS";
054            /**
055             * All characters are either letters or digits
056             */
057            public static final String ALPHANUMERIC = "ALPHANUMERIC";
058            /**
059             * Contains some digits and some non-letters - but can still contain letters
060             */
061            public static final String SOME_DIGITS = "SOME_DIGITS";
062            /**
063             * matches a regular expression for Roman numerals.   
064             */
065            public static final String ROMAN_NUMERAL = "ROMAN_NUMERAL";
066            
067            static Pattern digitsPattern = Pattern.compile("[0-9]+");
068            static Pattern yearDigitsPattern = Pattern.compile("(?:1[0-9]{3,3})|(?:2[0|1][0-9]{2,2})");
069            static Pattern alphanumericPattern = Pattern.compile("[a-zA-Z0-9-]+");
070            static Pattern someLetters = Pattern.compile("[a-zA-Z]");
071            static Pattern someDigits = Pattern.compile("[0-9]");
072            static Pattern romanNumeralPattern = Pattern.compile("^M?M?M?(CM|CD|D?C?C?C?)(XC|XL|L?X?X?X?)(IX|IV|V?I?I?I?)$");
073            
074            public NumericTypeProliferator() {
075                    super(NumericTypeProliferator.DEFAULT_NAME);
076            }
077            public NumericTypeProliferator(String featureName) {
078                    super(featureName);
079            }
080            
081            /**
082             * If the value of the feature is a StringValue and is determined to be one of 
083             * DIGITS, YEAR_DIGITS, ALPHANUMERIC, SOME_DIGITS, or ROMAN_NUMERAL, then
084             * a feature containing one of those five values is returned.  If
085             * the value of the feature cannot be characterized by one of these
086             * five values, then an empty list is returned (e.g. the value is an empty string, contains only
087             * white space, or contains only letters, etc.)
088             * 
089             * <p>
090             * This method draws heavily from NumericTypeTagger.py written by Steven Bethard.  That code credits
091             * <a href="http://diveintopython.org/unit_testing/stage_5.html">Dive Into Python</a> for the regular
092             * expression for matching roman numerals.
093             * 
094             * @return a feature that has a value that is
095             * one of DIGITS, YEAR_DIGITS, ALPHANUMERIC, SOME_DIGITS, or ROMAN_NUMERAL.  Otherwise an empty list is 
096             * returned.  
097             */
098            
099            @Override
100            public List<Feature> proliferate(Feature feature) {
101                    String featureName = Feature.createName(this.getFeatureName(), feature.getName());
102                    Object featureValue = feature.getValue();
103                    if(featureValue == null) return Collections.emptyList();
104                    else if(featureValue instanceof String) {
105                            String value = featureValue.toString();
106                            if(value == null || value.length() == 0) return Collections.emptyList();
107    
108                            if(digitsPattern.matcher(value).matches()) {
109                                    if(yearDigitsPattern.matcher(value).matches()) {
110                                            return Collections.singletonList(new Feature(featureName, YEAR_DIGITS));
111                                    }
112                                    else return Collections.singletonList(new Feature(featureName, DIGITS));
113                            }
114                            else if(someDigits.matcher(value).find()) {
115                                    if(alphanumericPattern.matcher(value).matches() && someLetters.matcher(value).find()) {
116                                            return Collections.singletonList(new Feature(featureName, ALPHANUMERIC)); 
117                                    }
118                                    else return Collections.singletonList(new Feature(featureName, SOME_DIGITS));
119                            }
120                            else if(romanNumeralPattern.matcher(value).matches()) {
121                                    return Collections.singletonList(new Feature(featureName, ROMAN_NUMERAL));
122                            }
123                    }
124                    return Collections.emptyList();
125            }
126    }
