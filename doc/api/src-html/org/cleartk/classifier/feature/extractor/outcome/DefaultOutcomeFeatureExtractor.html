028    import java.util.Collections;
029    import java.util.List;

036    import org.cleartk.util.UIMAUtil;

046    public class DefaultOutcomeFeatureExtractor implements OutcomeFeatureExtractor{
047    
048            private static final long serialVersionUID = 7476684786572310025L;
049    
050            /**
051             * "org.cleartk.classifier.feature.extractor.outcome.DefaultOutcomeFeatureExtractor.PARAM_MOST_RECENT_OUTCOME"
052             * is a single, optional, integer parameter, defaulting to 1, that indicates the
053             * position of the first outcome to include. For example, the default value of 1
054             * means that if the outcomes produced so far by the classifier were [A, B, C, D],
055             * then the first outcome to be used as a feature would be D.
056             */
057            public static final String PARAM_MOST_RECENT_OUTCOME = "org.cleartk.classifier.feature.extractor.outcome.DefaultOutcomeFeatureExtractor.PARAM_MOST_RECENT_OUTCOME";
058            
059            /**
060             * "org.cleartk.classifier.feature.extractor.outcome.DefaultOutcomeFeatureExtractor.PARAM_LEAST_RECENT_OUTCOME"
061             * is a single, optional, integer parameter, defaulting to 3, that indicates the
062             * position of the last outcome to include. For example, the default value of 3
063             * means that if the outcomes produced so far by the classifier were [A, B, C, D],
064             * then the last outcome to be used as a feature would be B.
065             */
066            public static final String PARAM_LEAST_RECENT_OUTCOME = "org.cleartk.classifier.feature.extractor.outcome.DefaultOutcomeFeatureExtractor.PARAM_LEAST_RECENT_OUTCOME";
067            
068            /**
069             * "org.cleartk.classifier.feature.extractor.outcome.DefaultOutcomeFeatureExtractor.PARAM_USE_BIGRAM"
070             * is a single, optional, boolean parameter, defaulting to true, that when true
071             * indicates that bigrams of outcomes should be included as features.
072             */
073            public static final String PARAM_USE_BIGRAM = "org.cleartk.classifier.feature.extractor.outcome.DefaultOutcomeFeatureExtractor.PARAM_USE_BIGRAM";
074            
075            /**
076             * "org.cleartk.classifier.feature.extractor.outcome.DefaultOutcomeFeatureExtractor.PARAM_USE_TRIGRAM"
077             * is a single, optional, boolean parameter, defaulting to true, that when true
078             * indicates that trigrams of outcomes should be included as features.
079             */
080            public static final String PARAM_USE_TRIGRAM = "org.cleartk.classifier.feature.extractor.outcome.DefaultOutcomeFeatureExtractor.PARAM_USE_TRIGRAM";
081    
082            /**
083             * "org.cleartk.classifier.feature.extractor.outcome.DefaultOutcomeFeatureExtractor.PARAM_USE_4GRAM"
084             * is a single, optional, boolean parameter, defaulting to true, that when true
085             * indicates that 4-grams of outcomes should be included as features.
086             */
087            public static final String PARAM_USE_4GRAM = "org.cleartk.classifier.feature.extractor.outcome.DefaultOutcomeFeatureExtractor.PARAM_USE_4GRAM";
088    
089            private int mostRecentOutcome;
090            private int leastRecentOutcome;
091            private boolean useBigram;
092            private boolean useTrigram;
093            private boolean use4gram;
094            
095            public void initialize(UimaContext context) throws ResourceInitializationException {
096                    mostRecentOutcome = (Integer) UIMAUtil.getDefaultingConfigParameterValue(context, PARAM_MOST_RECENT_OUTCOME, 1);
097                    leastRecentOutcome = (Integer) UIMAUtil.getDefaultingConfigParameterValue(context, PARAM_LEAST_RECENT_OUTCOME, 3);
098                    
099                    if(mostRecentOutcome < 1) {
100                            throw new ResourceInitializationException(new IllegalArgumentException(String.format("the parameter '%1$s' must be greater than 1.", PARAM_MOST_RECENT_OUTCOME)));
101                    }
102    
103                    if(leastRecentOutcome < mostRecentOutcome) {
104                            throw new ResourceInitializationException(new IllegalArgumentException(String.format("the parameter '%1$s' must be greater than the value for parameter '%2$s'.", PARAM_LEAST_RECENT_OUTCOME, PARAM_MOST_RECENT_OUTCOME)));
105                    }
106    
107                    useBigram = (Boolean) UIMAUtil.getDefaultingConfigParameterValue(context, PARAM_USE_BIGRAM, true);
108                    useTrigram = (Boolean) UIMAUtil.getDefaultingConfigParameterValue(context, PARAM_USE_TRIGRAM, true);
109                    use4gram = (Boolean) UIMAUtil.getDefaultingConfigParameterValue(context, PARAM_USE_4GRAM, false);
110            }
111    
112            public List<Feature> extractFeatures(List<Object> previousOutcomes) {
113                    if(previousOutcomes == null || previousOutcomes.size() == 0) {
114                            return Collections.emptyList();
115                    }
116                    
117                    List<Feature> features = new ArrayList<Feature>();
118                    
119                    for(int i=mostRecentOutcome; i<=leastRecentOutcome; i++) {
120                            int index = previousOutcomes.size() - i;
121                            if(index >= 0) {
122                                    Feature feature = new WindowFeature("PreviousOutcome", previousOutcomes.get(index),WindowFeature.ORIENTATION_LEFT,i, (Feature)null);
123                                    features.add(feature);
124                            }
125                    }
126    
127                    if(useBigram && previousOutcomes.size() >= 2) {
128                            int size = previousOutcomes.size();
129                            String featureValue = previousOutcomes.get(size - 1).toString() +"_"+previousOutcomes.get(size-2);
130                            Feature feature = new WindowNGramFeature("PreviousOutcomes", featureValue, WindowNGramFeature.ORIENTATION_LEFT,
131                                            WindowNGramFeature.DIRECTION_LEFT_TO_RIGHT, "_", 2, 1, (List<Feature>)null );
132                            features.add(feature);
133    
134                    }
135    
136                    if(useTrigram && previousOutcomes.size() >= 3) {
137                            int size = previousOutcomes.size();
138                            String featureValue = previousOutcomes.get(size - 1).toString() +"_"+previousOutcomes.get(size-2)+"_"+previousOutcomes.get(size-3);
139                            Feature feature = new WindowNGramFeature("PreviousOutcomes", featureValue, WindowNGramFeature.ORIENTATION_LEFT,
140                                            WindowNGramFeature.DIRECTION_LEFT_TO_RIGHT, "_", 3, 1, (List<Feature>)null );
141                            features.add(feature);
142    
143                    }
144    
145                    if(use4gram && previousOutcomes.size() >= 4) {
146                            int size = previousOutcomes.size();
147                            String featureValue = previousOutcomes.get(size - 1).toString() +"_"+previousOutcomes.get(size-2)+"_"+previousOutcomes.get(size-3)+"_"+previousOutcomes.get(size-4);
148                            Feature feature = new WindowNGramFeature("PreviousOutcomes", featureValue, WindowNGramFeature.ORIENTATION_LEFT,
149                                            WindowNGramFeature.DIRECTION_LEFT_TO_RIGHT, "_", 4, 1, (List<Feature>)null );
150                            features.add(feature);
151                    }
152    
153                    return features;
154            }
