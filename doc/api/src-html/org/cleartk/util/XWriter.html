031    package org.cleartk.util;
032    
033    import java.io.File;
034    import java.io.FileOutputStream;
035    import java.io.IOException;
036    
037    import org.apache.uima.UimaContext;
038    import org.apache.uima.analysis_component.JCasAnnotator_ImplBase;
039    import org.apache.uima.analysis_engine.AnalysisEngineProcessException;
040    import org.apache.uima.cas.CAS;
041    import org.apache.uima.cas.impl.XCASSerializer;
042    import org.apache.uima.cas.impl.XmiCasSerializer;
043    import org.apache.uima.jcas.JCas;
044    import org.apache.uima.resource.ResourceInitializationException;
045    import org.apache.uima.util.XMLSerializer;
046    import org.xml.sax.SAXException;
047    
048    /**
049     * 
050     * A simple CAS consumer that generates XCAS (XML representation of the CAS)
051     * files in the filesystem.
052     * 
053     * @author Philip Ogren
054     */
055    
056    public class XWriter extends JCasAnnotator_ImplBase {
057            /**
058             * "OutputDirectory" is a single, required string parameter that takes a
059             * path to directory into which output files will be written.
060             */
061            public static final String PARAM_OUTPUT_DIRECTORY = "OutputDirectory";
062    
063            /**
064             * "XMLScheme" is a single, optional, string parameter that specifies the
065             * UIMA XML serialization scheme that should be used. Valid values for this
066             * parameter are "XMI" (default) and "XCAS".
067             * 
068             * @see XmiCasSerializer
069             * @see XCASSerializer
070             */
071            public static final String PARAM_XML_SCHEME = "XMLScheme";
072    
073            public static final String XMI = "XMI";
074    
075            public static final String XCAS = "XCAS";
076    
077            private File outputDirectory;
078    
079            private boolean useXMI = true;
080    
081            @Override
082            public void initialize(UimaContext context) throws ResourceInitializationException {
083                    super.initialize(context);
084                    outputDirectory = new File((String) context.getConfigParameterValue(PARAM_OUTPUT_DIRECTORY));
085                    if (!outputDirectory.exists()) {
086                            outputDirectory.mkdirs();
087                    }
088    
089                    String xmlScheme = (String) UIMAUtil.getDefaultingConfigParameterValue(context, PARAM_XML_SCHEME, XMI);
090    
091                    if (xmlScheme.equals(XMI)) useXMI = true;
092                    else if (xmlScheme.equals(XCAS)) useXMI = false;
093                    else throw new ResourceInitializationException(String.format(
094                                    "parameter '%1$s' must be either '%2$s' or '%3$s' or left empty.", PARAM_XML_SCHEME, XMI, XCAS), null);
095    
096            }
097    
098            @Override
099            public void process(JCas jcas) throws AnalysisEngineProcessException {
100                    String id = DocumentUtil.getIdentifier(jcas);
101                    try {
102                            if (useXMI) writeXmi(jcas.getCas(), id);
103                            else writeXCas(jcas.getCas(), id);
104                    }
105                    catch (IOException e) {
106                            throw new AnalysisEngineProcessException(e);
107                    }
108                    catch (SAXException e) {
109                            throw new AnalysisEngineProcessException(e);
110                    }
111            }
112    
113            private void writeXCas(CAS aCas, String id) throws IOException, SAXException {
114                    File outFile = new File(outputDirectory, id + ".xcas");
115                    FileOutputStream out = null;
116                    try {
117                            out = new FileOutputStream(outFile);
118                            XCASSerializer ser = new XCASSerializer(aCas.getTypeSystem());
119                            XMLSerializer xmlSer = new XMLSerializer(out, false);
120                            ser.serialize(aCas, xmlSer.getContentHandler());
121                    }
122                    finally {
123                            if (out != null) {
124                                    out.close();
125                            }
126                    }
127            }
128    
129            private void writeXmi(CAS aCas, String id) throws IOException, SAXException {
130                    File outFile = new File(outputDirectory, id + ".xmi");
131                    FileOutputStream out = null;
132    
133                    try {
134                            out = new FileOutputStream(outFile);
135                            XmiCasSerializer ser = new XmiCasSerializer(aCas.getTypeSystem());
136                            XMLSerializer xmlSer = new XMLSerializer(out, false);
137                            ser.serialize(aCas, xmlSer.getContentHandler());
138                    }
139                    finally {
140                            if (out != null) {
141                                    out.close();
142                            }
143                    }
144            }
145    
146    }
