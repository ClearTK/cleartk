031    package org.cleartk.util;
032    
033    import java.io.File;
034    import java.io.FileOutputStream;
035    import java.io.IOException;
036    
037    import org.apache.uima.UimaContext;
038    import org.apache.uima.analysis_component.JCasAnnotator_ImplBase;
039    import org.apache.uima.analysis_engine.AnalysisEngineDescription;

041    import org.apache.uima.cas.CAS;
042    import org.apache.uima.cas.impl.XCASSerializer;
043    import org.apache.uima.cas.impl.XmiCasSerializer;
044    import org.apache.uima.jcas.JCas;
045    import org.apache.uima.resource.ResourceInitializationException;
046    import org.apache.uima.util.XMLSerializer;

048    import org.uutuc.factory.AnalysisEngineFactory;
049    import org.xml.sax.SAXException;

051    /**
052     * 
053     * A simple CAS consumer that generates XCAS (XML representation of the CAS)
054     * files in the filesystem.
055     * 
056     * @author Philip Ogren
057     */
058    
059    public class XWriter extends JCasAnnotator_ImplBase {
060            /**
061             * "org.cleartk.util.XWriter.PARAM_OUTPUT_DIRECTORY"
062             * is a single, required string parameter that takes a
063             * path to directory into which output files will be written.
064             */
065            public static final String PARAM_OUTPUT_DIRECTORY = "org.cleartk.util.XWriter.PARAM_OUTPUT_DIRECTORY";
066    
067            /**
068             * "org.cleartk.util.XWriter.PARAM_XML_SCHEME"
069             * is a single, optional, string parameter that specifies the
070             * UIMA XML serialization scheme that should be used. Valid values for this
071             * parameter are "XMI" (default) and "XCAS".
072             * 
073             * @see XmiCasSerializer
074             * @see XCASSerializer
075             */
076            public static final String PARAM_XML_SCHEME = "org.cleartk.util.XWriter.PARAM_XML_SCHEME";
077    
078            public static final String XMI = "XMI";
079    
080            public static final String XCAS = "XCAS";
081    
082            private File outputDirectory;
083    
084            private boolean useXMI = true;
085    
086            @Override
087            public void initialize(UimaContext context) throws ResourceInitializationException {
088                    super.initialize(context);
089                    outputDirectory = new File((String) context.getConfigParameterValue(PARAM_OUTPUT_DIRECTORY));
090                    if (!outputDirectory.exists()) {
091                            outputDirectory.mkdirs();
092                    }
093    
094                    String xmlScheme = (String) UIMAUtil.getDefaultingConfigParameterValue(context, PARAM_XML_SCHEME, XMI);
095    
096                    if (xmlScheme.equals(XMI)) useXMI = true;
097                    else if (xmlScheme.equals(XCAS)) useXMI = false;
098                    else throw new ResourceInitializationException(String.format(
099                                    "parameter '%1$s' must be either '%2$s' or '%3$s' or left empty.", PARAM_XML_SCHEME, XMI, XCAS), null);
100    
101            }
102    
103            @Override
104            public void process(JCas jcas) throws AnalysisEngineProcessException {
105                    String id = ViewURIUtil.getURI(jcas);
106                    int index = Math.max(id.lastIndexOf("/"), id.lastIndexOf("\\")) + 1;
107                    id = id.substring(index);
108                    try {
109                            if (useXMI) writeXmi(jcas.getCas(), id);
110                            else writeXCas(jcas.getCas(), id);
111                    }
112                    catch (IOException e) {
113                            throw new AnalysisEngineProcessException(e);
114                    }
115                    catch (SAXException e) {
116                            throw new AnalysisEngineProcessException(e);
117                    }
118            }
119    
120            private void writeXCas(CAS aCas, String id) throws IOException, SAXException {
121                    File outFile = new File(outputDirectory, id + ".xcas");
122                    FileOutputStream out = null;
123                    try {
124                            out = new FileOutputStream(outFile);
125                            XCASSerializer ser = new XCASSerializer(aCas.getTypeSystem());
126                            XMLSerializer xmlSer = new XMLSerializer(out, false);
127                            ser.serialize(aCas, xmlSer.getContentHandler());
128                    }
129                    finally {
130                            if (out != null) {
131                                    out.close();
132                            }
133                    }
134            }
135    
136            private void writeXmi(CAS aCas, String id) throws IOException, SAXException {
137                    File outFile = new File(outputDirectory, id + ".xmi");
138                    FileOutputStream out = null;
139    
140                    try {
141                            out = new FileOutputStream(outFile);
142                            XmiCasSerializer ser = new XmiCasSerializer(aCas.getTypeSystem());
143                            XMLSerializer xmlSer = new XMLSerializer(out, false);
144                            ser.serialize(aCas, xmlSer.getContentHandler());
145                    }
146                    finally {
147                            if (out != null) {
148                                    out.close();
149                            }
150                    }
151            }
152    
153            public static AnalysisEngineDescription getDescription(String outputDirectory) throws ResourceInitializationException {
154                    return AnalysisEngineFactory.createPrimitiveAnalysisEngineDescription(
155                                    XWriter.class, ClearTKComponents.TYPE_SYSTEM_DESCRIPTION, 
156                                    ClearTKComponents.TYPE_PRIORITIES,
157                                    XWriter.PARAM_OUTPUT_DIRECTORY, outputDirectory);
158            }
159    }
