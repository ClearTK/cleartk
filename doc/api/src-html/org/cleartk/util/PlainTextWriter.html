049    package org.cleartk.util;
050    
051    import java.io.File;
052    import java.io.IOException;
053    
054    import org.apache.uima.UimaContext;
055    import org.apache.uima.analysis_component.JCasAnnotator_ImplBase;
056    import org.apache.uima.analysis_engine.AnalysisEngineProcessException;
057    import org.apache.uima.jcas.JCas;
058    import org.apache.uima.resource.ResourceInitializationException;
059    import org.apache.uima.util.FileUtils;
060    
061    
062    /**
063     * <br>Copyright (c) 2007-2008, Regents of the University of Colorado 
064     * <br>All rights reserved.
065    
066     *
067     *
068     * A simple CAS consumer that creates plain text files from the document text 
069     * given to each CAS
070     * 
071     * @author Philip Ogren
072     */
073    
074    public class PlainTextWriter extends JCasAnnotator_ImplBase
075    {
076            /** "OutputDirectory" is a single, required string parameter that takes a
077             *  path to directory into which output files will be written.
078             */
079            public static final String PARAM_OUTPUT_DIRECTORY = "OutputDirectory";
080    
081            private File outputDir;
082    
083            @Override
084            public void initialize(UimaContext context) throws ResourceInitializationException {
085                    super.initialize(context);
086                    this.outputDir = new File((String)UIMAUtil.getRequiredConfigParameterValue(
087                                    context, PlainTextWriter.PARAM_OUTPUT_DIRECTORY));
088                    if (!this.outputDir.exists()) {
089                            this.outputDir.mkdirs();
090                    }
091            }
092    
093            @Override
094            public void process(JCas jCas) throws AnalysisEngineProcessException {
095            String id = DocumentUtil.getIdentifier(jCas);
096                File outFile = new File(this.outputDir, id + ".txt");
097                try {
098                            FileUtils.saveString2File(jCas.getDocumentText(), outFile);
099                    } catch (IOException e) {
100                            throw new AnalysisEngineProcessException(e);
101                    }
102            }
103    }
