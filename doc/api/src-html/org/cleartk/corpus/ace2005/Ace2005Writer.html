058    public class Ace2005Writer extends JCasAnnotator_ImplBase
059    {
060            /**
061             * "org.cleartk.corpus.ace2005.Ace2005Writer.PARAM_OUTPUT_DIRECTORY"
062             * is a single, required, string parameter that provides the path of the
063             * directory where the XML files should be written.
064             */
065            public static final String PARAM_OUTPUT_DIRECTORY = "org.cleartk.corpus.ace2005.Ace2005Writer.PARAM_OUTPUT_DIRECTORY";
066    
067            private File outputDir;
068            private int idIndex = 0;
069    
070            @Override
071            public void initialize(UimaContext context) throws ResourceInitializationException {
072                    super.initialize(context);
073                    outputDir = new File((String)UIMAUtil.getRequiredConfigParameterValue(
074                                    context, Ace2005Writer.PARAM_OUTPUT_DIRECTORY));
075                    if (!outputDir.exists())
076                    {
077                            outputDir.mkdirs();
078                    }
079            }
080    
081            private Element createExtentElement(String elementName, SimpleAnnotation annotation) {
082                    Element extent = new Element(elementName);
083                    Element charseq = new Element("charseq");
084                    charseq.setAttribute("START", ""+annotation.getBegin());
085                    charseq.setAttribute("END", ""+(annotation.getEnd()-1));
086                    charseq.setText(annotation.getCoveredText());
087                    extent.addContent(charseq);
088                    return extent;
089            }
090            
091            @Override
092            public void process(JCas jCas) throws AnalysisEngineProcessException {
093                    String uri = ViewURIUtil.getURI(jCas);
094                    String docId = uri.substring(0, uri.indexOf(".sgm"));
095                    org.cleartk.corpus.ace2005.type.Document document;
096                    document = AnnotationRetrieval.getAnnotations(jCas,
097                                    org.cleartk.corpus.ace2005.type.Document.class).iterator().next();
098                    
099                    Document xml = new Document();
100                    
101                    Element sourceFileElement = new Element("source_file");
102                    sourceFileElement.setAttribute("URI", uri);
103                    sourceFileElement.setAttribute("SOURCE", document.getAceSource());
104                    sourceFileElement.setAttribute("TYPE", document.getAceType());
105                    xml.addContent(sourceFileElement);
106                    
107                    Element documentElement = new Element("document");
108                    documentElement.setAttribute("DOCID", docId);
109                    sourceFileElement.addContent(documentElement);
110                    
111                    FSIterator namedEntities = jCas.getFSIndexRepository().getAllIndexedFS(jCas.getCasType(NamedEntity.type));
112                    while(namedEntities.hasNext()) {
113                            NamedEntity namedEntity = (NamedEntity) namedEntities.next();
114                            Element namedEntityElement = new Element("entity");
115                            namedEntityElement.setAttribute("ID", ""+idIndex++);
116                            namedEntityElement.setAttribute("TYPE", namedEntity.getEntityType());
117                            String entitySubtype = namedEntity.getEntitySubtype();
118                            if(entitySubtype != null)
119                                    namedEntityElement.setAttribute("SUBTYPE", entitySubtype);
120                            String entityClass = namedEntity.getEntityClass();
121                            if(entityClass != null)
122                                    namedEntityElement.setAttribute("CLASS", entityClass);
123                            
124                            FSArray namedEntityMentions = namedEntity.getMentions();
125                            for (int i=0; i<namedEntityMentions.size(); i++) {
126                                    NamedEntityMention namedEntityMention = (NamedEntityMention) namedEntityMentions.get(i);
127                                    Element namedEntityMentionElement = new Element("entity_mention");
128                                    namedEntityMentionElement.setAttribute("ID", ""+idIndex++);
129                                    namedEntityMentionElement.setAttribute("TYPE", namedEntityMention.getMentionType());
130                                    
131                                    namedEntityMentionElement.addContent(createExtentElement("extent", namedEntityMention.getAnnotation()));
132                                    namedEntityMentionElement.addContent(createExtentElement("head", namedEntityMention.getHead()));
133                                    namedEntityElement.addContent(namedEntityMentionElement);
134                            }
135                            documentElement.addContent(namedEntityElement);
136                    }
137                    
138            XMLOutputter xmlOut = new XMLOutputter(Format.getPrettyFormat());
139            try {
140                    FileOutputStream stream = new FileOutputStream(new File(outputDir, docId+".cleartk.xml"));
141                            xmlOut.output(xml, stream);
142                    stream.close();
143                    } catch (IOException e) {
144                            throw new AnalysisEngineProcessException(e);
145                    }
146                    
147            }
