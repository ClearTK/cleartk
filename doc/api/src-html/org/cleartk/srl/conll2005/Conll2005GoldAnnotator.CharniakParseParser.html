062            @Override
063            public void process(JCas jCas) throws AnalysisEngineProcessException {
064                    try {
065                            JCas conllView = jCas.getView("CoNLL2005View");
066                            JCas initView = jCas.getView("_InitialView");
067                            
068                            String conllText = conllView.getSofaDataString();
069    
070                            List<CoNLL2005Line> conll2005Lines = new ArrayList<CoNLL2005Line>();
071                            for( String line : conllText.split("\n") )
072                                    conll2005Lines.add(new CoNLL2005Line(line));
073                                                    
074                            
075                            StringBuffer docText = new StringBuffer();
076    
077                            List<TreebankNode> terminals = new ArrayList<TreebankNode>(conll2005Lines.size());
078                            CharniakParseParser parser = new CharniakParseParser(initView);
079                            
080                            int numberOfPredicates = conll2005Lines.get(0).argumentSegments.length;
081                            int currentPredicate = 0;
082                            PredicateParser predicateParsers[] = new PredicateParser[numberOfPredicates];
083                            for( int i = 0; i < numberOfPredicates; i++ )
084                                    predicateParsers[i] = new PredicateParser(initView);
085    
086                            for( CoNLL2005Line line : conll2005Lines.toArray(new CoNLL2005Line[0])) {
087                                    Assert.assertEquals(line.argumentSegments.length, numberOfPredicates);
088                                    
089                                    if( docText.length() > 0 && line.word.length() > 0 )
090                                            docText.append(" ");
091                                    int startIndex = docText.length();
092                                    docText.append(line.word);
093                                    int endIndex = docText.length();
094    
095                                    Token token = new Token(initView, startIndex, endIndex);
096                                    token.setPos(line.pos);
097                                    token.addToIndexes();
098                                    
099                                    TreebankNode terminal = new TreebankNode(initView, startIndex, endIndex);
100                                    terminal.setNodeType(line.pos);
101                                    terminal.setNodeValue(line.word);
102                                    terminal.setChildren(UIMAUtil.toFSArray(jCas, Collections.<TreebankNode>emptyList()));
103                                    terminal.setLeaf(true);
104                                    terminal.addToIndexes();
105                                    terminals.add(terminal);
106                                    
107                                    parser.feed(line.charniakParseSegment, terminal);
108                                    
109                                    for( int i = 0; i < numberOfPredicates; i++ ) {
110                                            predicateParsers[i].feed(line.argumentSegments[i], token);
111                                    }
112                                    
113                                    if( ! line.verbBaseForm.equals("-") ) {
114                                            predicateParsers[currentPredicate].feedInfo(line.word, line.verbBaseForm, line.verbSenseTag, token);
115                                            currentPredicate += 1;
116                                    }
117                            }
118                            initView.setSofaDataString(docText.toString(), "text/plain");
119                            
120                            Document conllDoc = DocumentUtil.getDocument(conllView);
121                            DocumentUtil.createDocument(initView, conllDoc.getIdentifier(), conllDoc.getPath());
122                            
123                            Sentence sentence = new Sentence(initView, 0, docText.toString().length());
124    
125                            TopTreebankNode constituentParse = parser.makeParse();
126                            sentence.setConstituentParse(constituentParse);
127                            sentence.addToIndexes();
128                            
129                            for( PredicateParser predicateParser : predicateParsers )
130                                    predicateParser.makePredicate();
131                            
132                    } catch (CASException e) {
133                            throw new AnalysisEngineProcessException(e);
134                    } catch (IOException e) {
135                            throw new AnalysisEngineProcessException(e);
136                    }
137            }

139            private static class CoNLL2005Line {
140                    String word;
141                    String pos;
142                    String charniakParseSegment;
143                    String neSegment;
144                    String verbSenseTag;
145                    String verbBaseForm;
146                    String argumentSegments[];
147                    
148                    CoNLL2005Line(String line) {
149                            String fields[] = line.split("\\s+");
150                            this.word = fields[0].trim();
151                            this.pos = fields[1].trim();
152                            this.charniakParseSegment = fields[2].trim();
153                            this.neSegment = fields[3].trim();
154                            this.verbSenseTag = fields[4].trim();
155                            this.verbBaseForm = fields[5].trim();
156                            this.argumentSegments = new String[fields.length - 6];
157                            for( int i = 6; i < fields.length; i++ )
158                                    this.argumentSegments[i-6] = fields[i].trim();
159                    }
160            }

162            private static class Constituent {
163                    String type;
164                    List<TreebankNode> children;
165                    
166                    Constituent(String type) {
167                            this.type = type;
168                            this.children = new ArrayList<TreebankNode>();
169                    }
170                    
171                    Constituent() {
172                            this(null);
173                    }
174                    
175                    public void addChild(TreebankNode newChild) {
176                            this.children.add(newChild);
177                    }
178                    
179                    public TreebankNode makeTreebankNode(JCas jCas) {
180                            if( this.type.equals("S1") ) {
181                                    return this.children.get(0);
182                            } else {
183                                    Span span = AnnotationUtil.getAnnotationsExtent(this.children);
184                                    TreebankNode node = new TreebankNode(jCas, span.getStart(), span.getEnd());
185                                    node.setNodeType(this.type);
186                                    node.setChildren(UIMAUtil.toFSArray(jCas, this.children));
187                                    for( TreebankNode child : this.children )
188                                            child.setParent(node);
189                                    node.addToIndexes();
190                                    return node;
191                            }
192                    }
193            }
194    
195            private static class CharniakParseParser {
196                    Stack<Constituent> parseStack;
197                    List<TreebankNode> terminals;
198                    
199                    JCas jCas;
200                    
201                    CharniakParseParser(JCas jCas) {
202                            parseStack = new Stack<Constituent>();
203                            parseStack.push(new Constituent("TOP"));
204                            terminals = new ArrayList<TreebankNode>();
205                            this.jCas = jCas;
206                    }
207    
208                    void feed(String segment, TreebankNode terminal) throws IOException {
209                            BufferedReader r = new BufferedReader(new StringReader(segment));
210    
211                            terminals.add(terminal);
212    
213                            for( int i = r.read(); i != -1; i = r.read() ) {
214                                    char c = (char) i;
215                                    switch(c) {
216                                    case '*':
217                                            parseStack.peek().addChild(terminal);
218                                            break;
219                                    case '(':
220                                            parseStack.push(new Constituent(readNodeType(r)));
221                                            break;
222                                    case ')':
223                                            TreebankNode node = parseStack.pop().makeTreebankNode(jCas);
224                                            parseStack.peek().addChild(node);
225                                            break;
226                                    default:
227                                            throw new IOException("unexpected character in string: " + String.valueOf(c) + " (" + String.valueOf((int) c) + ")");
228                                    }
229                            }
230                    }
231                    
232                    public TopTreebankNode makeParse() {
233                            Span span = AnnotationUtil.getAnnotationsExtent(this.terminals);
234                            TopTreebankNode node = new TopTreebankNode(jCas, span.getStart(), span.getEnd());
235                            node.setNodeType("TOP");
236                            node.setChildren(UIMAUtil.toFSArray(jCas, parseStack.peek().children));
237                            for( TreebankNode child : parseStack.peek().children )
238                                    child.setParent(node);
239                            node.setTerminals(UIMAUtil.toFSArray(jCas, this.terminals));
240                            node.addToIndexes();
241                            parseStack.pop();
242                            return node;
243                    }
244                    
245                    private static String readNodeType(BufferedReader r) throws IOException {
246                            StringBuffer b = new StringBuffer();
247                            
248                            while( r.ready() ) {
249                                    r.mark(1);
250                                    char c = (char) r.read();
251                                    if( c == '(' || c == ')' || c == '*' ) {
252                                            r.reset();
253                                            break;
254                                    } else {
255                                            b.append(c);
256                                    }
257                            }
258                            
259                            return b.toString();
260                    }
261            }

263            private static class PredicateParser {
264                    JCas jCas;
265                    
266                    String token;
267                    String baseForm;
268                    String sense;
269                    Token predicateToken;
270                    
271                    List<SemanticArgument> arguments;
272                    List<Token> argumentTokens;
273                    String argumentType;
274                    
275                    PredicateParser(JCas jCas) {
276                            this.jCas = jCas;
277                            this.arguments = new ArrayList<SemanticArgument>();
278                    }
279                    
280                    void feedInfo(String tokenText, String baseForm, String sense, Token token) {
281                            Assert.assertNotNull(token);
282                            this.token = tokenText;
283                            this.baseForm = baseForm;
284                            this.sense = sense;
285                            this.predicateToken = token;
286                    }
287                    
288                    void feed(String segment, Token token) throws IOException {
289                            BufferedReader r = new BufferedReader(new StringReader(segment));
290                            
291                            for( int i = r.read(); i != -1; i = r.read() ) {
292                                    char c = (char) i;
293                                    
294                                    switch(c) {
295                                            case '(':
296                                                    this.argumentTokens = new ArrayList<Token>();
297                                                    this.argumentType = readArgumentType(r);
298                                                    break;
299                                            case ')':
300                                                    Span span = AnnotationUtil.getAnnotationsExtent(this.argumentTokens);
301                                                    SemanticArgument arg = new SemanticArgument(jCas, span.getStart(), span.getEnd());
302                                                    arg.addToIndexes();
303                                                    SimpleAnnotation relation = AnnotationRetrieval.getMatchingAnnotation(jCas, arg, TreebankNode.class);
304                                                    if( relation == null ) {
305                                                            Chunk chunk = new Chunk(jCas, span.getStart(), span.getEnd());
306                                                            chunk.setTokens(UIMAUtil.toFSArray(jCas, this.argumentTokens));
307                                                            relation = chunk;
308                                                    }
309                                                    arg.setAnnotation(relation);
310                                                    arg.setLabel(this.argumentType);
311                                                    arg.addToIndexes();
312                                                    this.arguments.add(arg);
313                                                    
314                                                    this.argumentTokens = null;
315                                                    break;
316                                            case '*':
317                                                    if( this.argumentTokens != null )
318                                                            this.argumentTokens.add(token);
319                                                    break;
320                                            default:
321                                                    throw new IOException("unexpected character in string: " + String.valueOf(c) + " (" + String.valueOf((int) c) + ")");
322                                    }
323                            }
324                    }
325                    
326                    Predicate makePredicate() {
327                            Assert.assertNotNull(this.predicateToken);
328                            Predicate predicate = new Predicate(jCas, this.predicateToken.getBegin(), this.predicateToken.getEnd());
329                            predicate.setAnnotation(this.predicateToken);
330                            predicate.setArguments(UIMAUtil.toFSArray(jCas, this.arguments));
331                            predicate.setBaseForm(this.baseForm);
332                            predicate.addToIndexes();
333                            
334                            return predicate;
335                    }
336                    
337                    private static String readArgumentType(BufferedReader r) throws IOException {
338                            StringBuffer b = new StringBuffer();
339                            
340                            while( true ) {
341                                    r.mark(1);
342                                    int i = r.read();
343                                    if( i == -1 )
344                                            break;
345                                    
346                                    char c = (char) i;
347                                    if( c == '(' || c == ')' || c == '*' ) {
348                                            r.reset();
349                                            break;
350                                    }
351                                    
352                                    b.append(c);
353                            }
354                            
355                            return b.toString();
356                    }
357            }
