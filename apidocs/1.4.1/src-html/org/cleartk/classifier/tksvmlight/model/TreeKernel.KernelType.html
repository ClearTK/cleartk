027import java.util.HashMap;
028import java.util.List;
029import java.util.concurrent.ConcurrentHashMap;
030
031import org.apache.commons.lang.NotImplementedException;
032import org.cleartk.classifier.tksvmlight.TreeFeatureVector;
033import org.cleartk.syntax.constituent.util.TopTreebankNode;
034import org.cleartk.corpus.penntreebank.TreebankFormatParser;
035import org.cleartk.syntax.constituent.util.TreebankNode;

044public class TreeKernel {
045  public static final double LAMBDA_DEFAULT = 0.4;
046
047  private double lambda = LAMBDA_DEFAULT;
048
049  private boolean normalize = false;
050
051  private ConcurrentHashMap<String, Double> normalizers = new ConcurrentHashMap<String, Double>();
052
053  public static enum ForestSumMethod {
054    SEQUENTIAL, ALL_PAIRS
055  }
056
057  private ForestSumMethod sumMethod = ForestSumMethod.SEQUENTIAL;
058
059  public static enum KernelType {
060    SUBSET, SUBTREE, SUBSET_BOW, PARTIAL
061  }
062
063  private KernelType kernelType;
064
065  HashMap<String, TopTreebankNode> trees = null;
066
067  public TreeKernel(
068      double lambda,
069      ForestSumMethod sumMethod,
070      KernelType kernelType,
071      boolean normalize) {
072    this.lambda = lambda;
073    this.sumMethod = sumMethod;
074    this.kernelType = kernelType;
075    this.normalize = normalize;
076    trees = new HashMap<String, TopTreebankNode>();
077  }
078
079  public double evaluate(TreeFeatureVector fv1, TreeFeatureVector fv2) {
080    double sim = 0.0;
081    if (sumMethod == ForestSumMethod.SEQUENTIAL) {
082      List<String> fv1Trees = new ArrayList<String>(fv1.getTrees().values());
083      List<String> fv2Trees = new ArrayList<String>(fv2.getTrees().values());
084      for (int i = 0; i < fv1Trees.size(); i++) {
085        String tree1Str = fv1Trees.get(i);
086        String tree2Str = fv2Trees.get(i);
087        if (kernelType == KernelType.SUBSET) {
088          sim += sst(tree1Str, tree2Str);
089        } else {
090          throw new NotImplementedException("The only kernel type implemented is SUBSET!");
091        }
092      }
093    } else {
094      throw new NotImplementedException("The only summation method implemented is Sequential!");
095    }
096    return sim;
097  }
098
099  private double sst(String tree1Str, String tree2Str) {
100
101    TopTreebankNode node1 = null;
102    if (!trees.containsKey(tree1Str)) {
103      node1 = TreebankFormatParser.parse(tree1Str);
104      trees.put(tree1Str, node1);
105    } else
106      node1 = trees.get(tree1Str);
107
108    TopTreebankNode node2 = null;
109    if (!trees.containsKey(tree2Str)) {
110      node2 = TreebankFormatParser.parse(tree2Str);
111      trees.put(tree2Str, node2);
112    } else
113      node2 = trees.get(tree2Str);
114
115    double norm1 = 0.0;
116    double norm2 = 0.0;
117    if (normalize) {
118      if (!normalizers.containsKey(tree1Str)) {
119        double norm = sim(node1, node1);
120        normalizers.put(tree1Str, norm);
121      }
122      if (!normalizers.containsKey(tree2Str)) {
123        double norm = sim(node2, node2);
124        normalizers.put(tree2Str, norm);
125      }
126
127      norm1 = normalizers.get(tree1Str);
128      norm2 = normalizers.get(tree2Str);
129    }
130    if (normalize) {
131      return (sim(node1, node2) / Math.sqrt(norm1 * norm2));
132    } else {
133      return sim(node1, node2);
134    }
135  }
136
137  private double sim(TreebankNode node1, TreebankNode node2) {
138    double sim = 0.0;
139    List<TreebankNode> N1 = getNodeList(node1);
140    List<TreebankNode> N2 = getNodeList(node2);
141    for (TreebankNode n1 : N1) {
142      for (TreebankNode n2 : N2) {
143        sim += numCommonSubtrees(n1, n2);
144      }
145    }
146    return sim;
147  }
148
149  private double numCommonSubtrees(TreebankNode n1, TreebankNode n2) {
150    double retVal = 1.0;
151    List<TreebankNode> children1 = n1.getChildren();
152    List<TreebankNode> children2 = n2.getChildren();
153    int c1size = children1.size();
154    int c2size = children2.size();
155    if (c1size != c2size) {
156      retVal = 0;
157    } else if (!n1.getType().equals(n2.getType())) {
158      retVal = 0;
159    } else if (n1.isLeaf() && n2.isLeaf()) {
160      // both are preterminals, and we know they have the same type, need to check value (word)
161      // Collins & Duffy tech report says lambdaSquared, but Nips 02 paper uses lambda
162      // Moschitti's papers also use lambda
163      // retVal = lambdaSquared;
164      if (n1.getValue().equals(n2.getValue())) {
165        retVal = lambda;
166      } else {
167        retVal = 0;
168      }
169    } else {
170      // At this point they have the same label and same # children. Check if children the same.
171      boolean sameProd = true;
172      for (int i = 0; i < c1size; i++) {
173        String l1 = children1.get(i).getType();
174        String l2 = children2.get(i).getType();
175        if (!l1.equals(l2)) {
176          sameProd = false;
177          break;
178        }
179      }
180      if (sameProd == true) {
181        for (int i = 0; i < c1size; i++) {
182          retVal *= (1 + numCommonSubtrees(children1.get(i), children2.get(i)));
183        }
184        // again, some disagreement in the literature, with Collins and Duffy saying
185        // lambdaSquared here in tech report and lambda here in nips 02. We'll stick with
186        // lambda b/c that's what moschitti's code (which was presumably used for model-building)
187        // uses.
188        retVal = lambda * retVal;
189      } else {
190        retVal = 0;
191      }
192    }
193    return retVal;
194  }
195
196  private static final List<TreebankNode> getNodeList(TreebankNode tree) {
197    ArrayList<TreebankNode> list = new ArrayList<TreebankNode>();
198    list.add(tree);
199    for (int i = 0; i < list.size(); ++i) {
200      list.addAll(list.get(i).getChildren());
201    }
202    return list;
203  }
