062  public static final String PARAM_OUTPUT_DIRECTORY_NAME = ConfigurationParameterFactory.createConfigurationParameterName(
063      TempEval2007Writer.class,

066  @ConfigurationParameter(
067      description = "Provides the path where the TimeML documents should be written.",
068      mandatory = true)
069  private String outputDirectoryName;
070
071  public static AnalysisEngineDescription getDescription(String outputDir)
072      throws ResourceInitializationException {
073    return AnalysisEngineFactory.createPrimitiveDescription(
074        TempEval2007Writer.class,
075        PARAM_OUTPUT_DIRECTORY_NAME,
076        outputDir);
077  }
078
079  private File outputDirectory;
080
081  @Override
082  public void initialize(UimaContext context) throws ResourceInitializationException {
083    super.initialize(context);
084
085    this.outputDirectory = new File(outputDirectoryName);
086    if (!this.outputDirectory.exists()) {
087      this.outputDirectory.mkdirs();
088    }
089  }
090
091  public static String toTimeML(JCas jCas) throws AnalysisEngineProcessException {
092    try {
093      return toXML(jCas.getCas(), new TimeMLAnnotationsToElements());
094    } catch (SAXException e) {
095      throw new AnalysisEngineProcessException(e);
096    }
097  }
098
099  @Override
100  public void process(JCas jCas) throws AnalysisEngineProcessException {
101    String xmlString = toTimeML(jCas);
102
103    // write the TimeML to the output file
104    String filePath = ViewURIUtil.getURI(jCas).getPath();
105    String fileName = new File(filePath).getName();
106    if (!fileName.endsWith(".tml")) {
107      fileName += ".tml";
108    }
109    File outputFile = new File(this.outputDirectory, fileName);
110    try {
111      FileWriter writer = new FileWriter(outputFile);
112      writer.write(xmlString);
113      writer.close();
114    } catch (IOException e) {
115      throw new AnalysisEngineProcessException(e);
116    }
117  }
118
119  public void setOutputDirectoryName(String outputDirectoryName) {
120    this.outputDirectoryName = outputDirectoryName;
121  }

123  private static interface AnnotationsToElements {
124    public void startRootElement(ContentHandler handler) throws SAXException;
125
126    public void endRootElement(ContentHandler handler) throws SAXException;
127
128    public void startAnnotationElement(AnnotationFS annotation, ContentHandler handler)
129        throws SAXException;
130
131    public void endAnnotationElement(AnnotationFS annotation, ContentHandler handler)
132        throws SAXException;
133  }
134
135  private static class TimeMLAnnotationsToElements implements AnnotationsToElements {
136
137    public TimeMLAnnotationsToElements() {
138    }
139
140    @Override
141    public void startRootElement(ContentHandler handler) throws SAXException {
142      handler.startElement("", "TimeML", "TimeML", new AttributesImpl());
143    }
144
145    @Override
146    public void endRootElement(ContentHandler handler) throws SAXException {
147      handler.endElement("", "TimeML", "TimeML");
148    }
149
150    @Override
151    public void startAnnotationElement(AnnotationFS annotation, ContentHandler handler)
152        throws SAXException {
153      String name = TimeMLUtil.toTimeMLElementName(annotation);
154      if (name != null) {
155        handler.startElement("", name, name, TimeMLUtil.toTempEval2007Attributes(annotation, name));
156      }
157    }
158
159    @Override
160    public void endAnnotationElement(AnnotationFS annotation, ContentHandler handler)
161        throws SAXException {
162      String name = TimeMLUtil.toTimeMLElementName(annotation);
163      if (name != null) {
164        handler.endElement("", name, name);
165      }
166    }
167  }

169  /**
170   * Copied and modified from {@link org.apache.uima.util.CasToInlineXml}
171   */
172  private static String toXML(CAS cas, AnnotationsToElements converter) throws SAXException {
173    ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
174    XMLSerializer sax2xml = new XMLSerializer(byteArrayOutputStream, false);
175
176    // get document text
177    String docText = cas.getDocumentText();
178    char[] docCharArray = docText.toCharArray();
179
180    // get iterator over annotations sorted by increasing start position and
181    // decreasing end position
182    FSIterator<AnnotationFS> iterator = cas.getAnnotationIndex().iterator();
183
184    // This is basically a recursive algorithm that has had the recursion
185    // removed through the use of an explicit Stack. We iterate over the
186    // annotations, and if an annotation contains other annotations, we
187    // push the parent annotation on the stack, process the children, and
188    // then come back to the parent later.
189    List<AnnotationFS> stack = new ArrayList<AnnotationFS>();
190    int pos = 0;
191
192    ContentHandler handler = sax2xml.getContentHandler();
193    handler.startDocument();
194    // write the start tag
195    converter.startRootElement(handler);
196    // now use null is a placeholder for this artificial Document annotation
197    AnnotationFS curAnnot = null;
198
199    while (iterator.isValid()) {
200      AnnotationFS nextAnnot = iterator.get();
201
202      if (curAnnot == null || nextAnnot.getBegin() < curAnnot.getEnd()) {
203        // nextAnnot's start point is within the span of curAnnot
204        if (curAnnot == null || nextAnnot.getEnd() <= curAnnot.getEnd()) // crossover span check
205        {
206          // nextAnnot is contained within curAnnot
207
208          // write text between current pos and beginning of nextAnnot
209          try {
210            handler.characters(docCharArray, pos, nextAnnot.getBegin() - pos);
211            pos = nextAnnot.getBegin();
212            converter.startAnnotationElement(nextAnnot, handler);
213
214            // push parent annotation on stack
215            stack.add(curAnnot);
216            // move on to next annotation
217            curAnnot = nextAnnot;
218          } catch (StringIndexOutOfBoundsException e) {
219            System.err.println("Invalid annotation range: " + nextAnnot.getBegin() + ","
220                + nextAnnot.getEnd() + " in document of length " + docText.length());
221          }
222        }
223        iterator.moveToNext();
224      } else {
225        // nextAnnot begins after curAnnot ends
226        // write text between current pos and end of curAnnot
227        try {
228          handler.characters(docCharArray, pos, curAnnot.getEnd() - pos);
229          pos = curAnnot.getEnd();
230        } catch (StringIndexOutOfBoundsException e) {
231          System.err.println("Invalid annotation range: " + curAnnot.getBegin() + ","
232              + curAnnot.getEnd() + " in document of length " + docText.length());
233        }
234        converter.endAnnotationElement(curAnnot, handler);
235
236        // pop next containing annotation off stack
237        curAnnot = stack.remove(stack.size() - 1);
238      }
239    }
240
241    // finished writing all start tags, now finish up
242    if (curAnnot != null) {
243      try {
244        handler.characters(docCharArray, pos, curAnnot.getEnd() - pos);
245        pos = curAnnot.getEnd();
246      } catch (StringIndexOutOfBoundsException e) {
247        System.err.println("Invalid annotation range: " + curAnnot.getBegin() + ","
248            + curAnnot.getEnd() + "in document of length " + docText.length());
249      }
250      converter.endAnnotationElement(curAnnot, handler);
251
252      while (!stack.isEmpty()) {
253        curAnnot = stack.remove(stack.size() - 1); // pop
254        if (curAnnot == null) {
255          break;
256        }
257        try {
258          handler.characters(docCharArray, pos, curAnnot.getEnd() - pos);
259          pos = curAnnot.getEnd();
260        } catch (StringIndexOutOfBoundsException e) {
261          System.err.println("Invalid annotation range: " + curAnnot.getBegin() + ","
262              + curAnnot.getEnd() + "in document of length " + docText.length());
263        }
264        converter.endAnnotationElement(curAnnot, handler);
265      }
266    }
267
268    if (pos < docCharArray.length) {
269      handler.characters(docCharArray, pos, docCharArray.length - pos);
270    }
271    converter.endRootElement(handler);
272    handler.endDocument();
273
274    // return XML string
275    return new String(byteArrayOutputStream.toByteArray());
276  }
