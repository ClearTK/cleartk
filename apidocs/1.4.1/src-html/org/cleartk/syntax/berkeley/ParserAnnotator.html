057public class ParserAnnotator<TOKEN_TYPE extends Annotation, SENTENCE_TYPE extends Annotation, TOP_NODE_TYPE extends Annotation>
058    extends ParserWrapper_ImplBase<TOKEN_TYPE, SENTENCE_TYPE, Tree<String>, TOP_NODE_TYPE> {
059  
060  public static AnalysisEngineDescription getDescription(String modelPath) throws ResourceInitializationException {
061    return AnalysisEngineFactory.createPrimitiveDescription(
062        ParserAnnotator.class,
063        ParserAnnotator.PARAM_PARSER_MODEL_PATH,
064        modelPath,
065        ParserWrapper_ImplBase.PARAM_OUTPUT_TYPES_HELPER_CLASS_NAME,
066        DefaultOutputTypesHelper.class.getName());
067  }
068
069  public static final String PARAM_PARSER_MODEL_PATH = ConfigurationParameterFactory.createConfigurationParameterName(
070      ParserAnnotator.class,
071      "parserModelPath");
072
073  @ConfigurationParameter
074  private String parserModelPath;
075
076  protected CoarseToFineMaxRuleParser parser;
077
078  private int parseFailureCount = 0;
079
080  private int sentenceCount = 0;
081
082  @Override
083  public void initialize(UimaContext context) throws ResourceInitializationException {
084    super.initialize(context);
085    System.out.print("initializing Berkeley Parser: " + parserModelPath + " ... ");
086
087    long start = System.nanoTime();
088
089    InputStream modelInputStream;
090    try {
091      modelInputStream = IOUtil.getInputStream(ParserAnnotator.class, parserModelPath);
092
093      GZIPInputStream gzis = new GZIPInputStream(modelInputStream); // Compressed
094      ObjectInputStream in = new ObjectInputStream(gzis); // Load objects
095      ParserData parserData = (ParserData) in.readObject(); // Read the mix of grammars
096
097      Grammar grammar = parserData.getGrammar();
098      Lexicon lexicon = parserData.getLexicon();
099      Numberer.setNumberers(parserData.getNumbs());
100      parser = new CoarseToFineMaxRuleParser(
101          grammar,
102          lexicon,
103          1.0,
104          -1,
105          false,
106          false,
107          false,
108          false,
109          false,
110          true,
111          true);
112      long stop = System.nanoTime();
113      float seconds = (float) (stop - start) / 1000000000;
114      System.out.println("done.   Loaded in: " + seconds + " seconds");
115    } catch (IOException e) {
116      throw new ResourceInitializationException(e);
117    } catch (ClassNotFoundException e) {
118      throw new ResourceInitializationException(e);
119    }
120
121  }
122
123  @Override
124  public void process(JCas jCas) throws AnalysisEngineProcessException {
125
126    List<SENTENCE_TYPE> sentenceList = inputTypesHelper.getSentences(jCas);
127
128    for (SENTENCE_TYPE sentence : sentenceList) {
129      sentenceCount++;
130      List<TOKEN_TYPE> tokens = inputTypesHelper.getTokens(jCas, sentence);
131      List<String> words = new ArrayList<String>();
132      List<String> tags = new ArrayList<String>();
133
134      for (TOKEN_TYPE token : tokens) {
135        words.add(token.getCoveredText());
136        String tag = inputTypesHelper.getPosTag(token);
137        tags.add(tag);
138      }
139
140      Tree<String> tree = parser.getBestConstrainedParse(words, tags, null);
141      if (tree.isLeaf()) {
142        System.out.println("words: " + words.size() + "  " + words);
143        System.out.println("tags: " + tags.size() + "  " + tags);
144        System.out.println("unable to parse sentence: " + sentence.getCoveredText());
145        parseFailureCount++;
146      } else {
147        outputTypesHelper.addParse(jCas, tree, sentence, tokens);
148      }
149    }
150  }
151
152  @Override
153  public void collectionProcessComplete() throws AnalysisEngineProcessException {
154    System.out.println("total number of sentences that were not parsed was: " + parseFailureCount
155        + " out of " + sentenceCount);
156  }
