056public class GrmmClassifierBuilder extends
057    SequenceClassifierBuilder_ImplBase<GrmmClassifier, List<NameNumber>, String[], String[]> {
058
059  private static String DEFAULT_MODEL_FILENAME = "acrf.model.ser.gz";
060
061  private static String JAR_ENTRY_MODEL = "model.grmm";
062
063  private static String JAR_ENTRY_OUTCOME_EXAMPLE = "dummy.outcome";
064
065  @Override
066  public File getTrainingDataFile(File dir) {
067    return new File(dir, "training-data.grmm");
068  }
069
070  /**
071   * Parameters:<br>
072   * <table border="1">
073   * <thead>
074   * <tr>
075   * <th>position</th>
076   * <th>type</th>
077   * <th>description</td></th>
078   * </tr>
079   * </thead>
080   * <tr>
081   * <td>0</td>
082   * <td>required</td>
083   * <td>path to template file</td>
084   * </tr>
085   * <tr>
086   * <td>1</td>
087   * <td>optional</td>
088   * <td>inferencer, see GRMM docs for possible values</td>
089   * </tr>
090   * <tr>
091   * <td></td>
092   * <td></td>
093   * <td>default: LoopyBP</td>
094   * </tr>
095   * <tr>
096   * <td>2</td>
097   * <td>optional</td>
098   * <td>max-inferencer, see GRMM docs for possible values</td>
099   * </tr>
100   * <tr>
101   * <td></td>
102   * <td></td>
103   * <td>default: LoopyBP.createForMaxProduct()</td>
104   * </tr>
105   * </table>
106   */
107  public void trainClassifier(File dir, String... args) throws Exception {
108    if (dir == null || !dir.isDirectory()) {
109      throw new IllegalArgumentException(String.format("invalid directory \"%s\"", dir));
110    }
111    if (args == null || args.length == 0 || args[0] == null) {
112      throw new IllegalArgumentException("missing template file in \"args\"");
113    }
114    File template = new File(dir, args[0]);
115    if (!template.exists()) {
116      String msg = "template file \"%s\" does not exist!";
117      throw new IllegalArgumentException(String.format(msg, template));
118    }
119    File outputFile = new File(dir, DEFAULT_MODEL_FILENAME);
120
121    String inferencer = args.length < 2 ? "LoopyBP" : args[1];
122    String maxInferencer = args.length < 3 ? "LoopyBP.createForMaxProduct()" : args[2];
123
124    // usage of GRMM:
125    String[] grmmArgs = new String[] {
126        "--training",
127        new File(dir, "training-data.grmm").getAbsolutePath(),
128        "--testing",
129        new File(dir, "training-data.grmm").getAbsolutePath(),
130        "--model-file",
131        template.getAbsolutePath(),
132        "--inferencer",
133        inferencer,
134        "--max-inferencer",
135        maxInferencer };
136
137    // GenericAcrfTui saves in the current directory; move to the appropriate directory
138    GenericAcrfTui.main(grmmArgs);
139    Files.move(new File("acrf.ser.gz"), outputFile.getAbsoluteFile());
140  }
141
142  @Override
143  protected void packageClassifier(File dir, JarOutputStream modelStream) throws IOException {
144    super.packageClassifier(dir, modelStream);
145
146    File model = new File(dir, DEFAULT_MODEL_FILENAME);
147    File trainingData = new File(dir, "training-data.grmm");
148    if (!model.exists()) {
149      String msg = "model file \"%s\" not found";
150      throw new IllegalArgumentException(String.format(msg, model));
151    }
152    if (!trainingData.exists()) {
153      String msg = "training data file \"%s\" not found";
154      throw new IllegalArgumentException(String.format(msg, trainingData));
155    }
156    // handle an outcome example from the training data over to the
157    // classifier
158    // through a special jar-file-entry:
159    LineNumberReader lnr = new LineNumberReader(new FileReader(trainingData));
160    this.outcomeExample = lnr.readLine().split("----")[0];
161    lnr.close();
162    JarStreams.putNextJarEntry(modelStream, JAR_ENTRY_MODEL, model);
163    modelStream.putNextEntry(new JarEntry(JAR_ENTRY_OUTCOME_EXAMPLE));
164    new ObjectOutputStream(modelStream).writeObject(outcomeExample);
165  }
166
167  protected ACRF acrf;
168
169  protected String outcomeExample;
170
171  @Override
172  protected void unpackageClassifier(JarInputStream modelStream) throws IOException {
173    super.unpackageClassifier(modelStream);
174    try {
175      JarStreams.getNextJarEntry(modelStream, JAR_ENTRY_MODEL);
176      this.acrf = (ACRF) new ObjectInputStream(new GZIPInputStream(modelStream)).readObject();
177      JarStreams.getNextJarEntry(modelStream, JAR_ENTRY_OUTCOME_EXAMPLE);
178      this.outcomeExample = (String) new ObjectInputStream(modelStream).readObject();
179    } catch (ClassNotFoundException e) {
180      throw new IOException(e);
181    }
182  }
183
184  @Override
185  protected GrmmClassifier newClassifier() {
186    return new GrmmClassifier(
187        this.featuresEncoder,
188        this.outcomeEncoder,
189        this.acrf,
190        this.outcomeExample);
191  }
