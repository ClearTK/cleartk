038import cc.mallet.fst.Transducer;

050public class MalletCRFStringOutcomeClassifierBuilder extends
051    SequenceClassifierBuilder_ImplBase<MalletCRFStringOutcomeClassifier, List<NameNumber>, String, String> {
052
053  private static final String MODEL_NAME = "model.malletcrf";
054
055  public File getTrainingDataFile(File dir) {
056    return new File(dir, "training-data.malletcrf");
057  }
058
059  public void trainClassifier(File dir, String... args) throws Exception {
060    String[] malletArgs = new String[args.length + 5];
061    System.arraycopy(args, 0, malletArgs, 0, args.length);
062    malletArgs[malletArgs.length - 5] = "--train";
063    malletArgs[malletArgs.length - 4] = "true";
064    malletArgs[malletArgs.length - 3] = "--model-file";
065    malletArgs[malletArgs.length - 2] = new File(dir, MODEL_NAME).getPath();
066    malletArgs[malletArgs.length - 1] = getTrainingDataFile(dir).getPath();
067    String leaveMyLoggingAloneMallet = "java.util.logging.config.file";
068    String propValue = System.getProperty(leaveMyLoggingAloneMallet);
069    System.setProperty(leaveMyLoggingAloneMallet, "anything-but-null");
070    try {
071      SimpleTagger.main(malletArgs);
072    } finally {
073      System.getProperties().remove(leaveMyLoggingAloneMallet);
074      if (propValue != null) {
075        System.setProperty(leaveMyLoggingAloneMallet, propValue);
076      }
077    }
078  }
079
080  @Override
081  protected void packageClassifier(File dir, JarOutputStream modelStream) throws IOException {
082    super.packageClassifier(dir, modelStream);
083    JarStreams.putNextJarEntry(modelStream, MODEL_NAME, new File(dir, MODEL_NAME));
084  }
085
086  protected Transducer transducer;
087
088  @Override
089  protected void unpackageClassifier(JarInputStream modelStream) throws IOException {
090    super.unpackageClassifier(modelStream);
091    JarStreams.getNextJarEntry(modelStream, MODEL_NAME);
092    ObjectInputStream objectStream = new ObjectInputStream(modelStream);
093    try {
094      this.transducer = (Transducer) objectStream.readObject();
095    } catch (ClassNotFoundException e) {
096      throw new IOException(e);
097    }
098  }
099
100  @Override
101  protected MalletCRFStringOutcomeClassifier newClassifier() {
102    return new MalletCRFStringOutcomeClassifier(this.featuresEncoder, this.outcomeEncoder, this.transducer);
103  }
104}
