049public class InstanceListCreator {
050
051  public static void main(String[] args) throws IOException {
052    String inputDataFileName = args[0];
053    String outputDataFileName = args[1];
054
055    InstanceListCreator instanceListCreator = new InstanceListCreator();
056    InstanceList instanceList = instanceListCreator.createInstanceList(new File(inputDataFileName));
057
058    instanceList.save(new File(outputDataFileName));
059  }
060
061  public InstanceList createInstanceList(File dataFile) throws IOException {
062
063    InstanceList instanceList = new InstanceList(new SerialPipes(new Pipe[] {
064        new Target2Label(),
065        new Csv2FeatureVector() }));
066
067    Reader fileReader = new FileReader(dataFile);
068    instanceList.addThruPipe(new DataIterator(fileReader));
069    fileReader.close();
070
071    return instanceList;
072
073  }
074
075  /**
076   * This DataIterator was cut-n-paste from cc.mallet.pipe.iterator.CsvIterator and modified
077   */
078  public class DataIterator implements Iterator<Instance> {
079
080    LineNumberReader reader;
081
082    String currentLine;
083
084    public DataIterator(Reader input) {
085      this.reader = new LineNumberReader(input);
086      try {
087        this.currentLine = reader.readLine();
088      } catch (IOException e) {
089        throw new IllegalStateException();
090      }
091    }
092
093    public Instance next() {
094      String data = null;
095      String target = null;
096
097      int split = currentLine.lastIndexOf(" ");
098      data = currentLine.substring(0, split);
099      target = currentLine.substring(split + 1);
100      Instance carrier = new Instance(data, target, null, null);
101
102      try {
103        this.currentLine = reader.readLine();
104      } catch (IOException e) {
105        throw new IllegalStateException();
106      }
107      return carrier;
108    }
109
110    public boolean hasNext() {
111      return currentLine != null;
112    }
113
114    public void remove() {
115      throw new IllegalStateException("This Iterator<Instance> does not support remove().");
116    }
117
118  }
119
