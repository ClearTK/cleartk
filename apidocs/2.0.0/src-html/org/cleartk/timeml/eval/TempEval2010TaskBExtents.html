024package org.cleartk.timeml.eval;
025
026import java.io.File;
027import java.util.ArrayList;
028import java.util.Arrays;
029import java.util.List;
030
031import org.cleartk.opennlp.tools.ParserAnnotator;
032import org.cleartk.opennlp.tools.PosTaggerAnnotator;
033import org.cleartk.corpus.timeml.TempEval2010GoldAnnotator;
034import org.cleartk.corpus.timeml.TempEval2010Writer;
035import org.cleartk.snowball.DefaultSnowballStemmer;
036import org.cleartk.timeml.event.EventAnnotator;
037import org.cleartk.timeml.type.Event;
038
039/**
040 * TempEval 2010 task B: event extents
041 * 
042 * Best reported token-level scores in TempEval 2010:
043 * <ul>
044 * <li>precision: 0.83</li>
045 * <li>recall: 0.88</li>
046 * <li>F1: 0.83</li>
047 * </ul>
048 * 
049 * <br>
050 * Copyright (c) 2011, Regents of the University of Colorado <br>
051 * All rights reserved.
052 * 
053 * @author Steven Bethard
054 */
055public class TempEval2010TaskBExtents extends TempEval2010Main {
056
057  public static void main(String[] args) throws Exception {
058    new TempEval2010TaskBExtents().runMain(args);
059  }
060
061  @Override
062  protected TempEval2010Evaluation getEvaluation(File trainDir, File testDir, File outputDir)
063      throws Exception {
064
065    List<ModelInfo<Event>> infos = new ArrayList<ModelInfo<Event>>();
066    infos.add(new EventModelInfo(null, EventAnnotator.FACTORY, new String[] { "MaxEnt" }));
067
068    return new TempEval2010Evaluation(
069        trainDir,
070        testDir,
071        outputDir,
072        Arrays.asList(TempEval2010GoldAnnotator.PARAM_TEXT_VIEWS),
073        TempEval2010GoldAnnotator.PARAM_EVENT_EXTENT_VIEWS,
074        TempEval2010Writer.PARAM_EVENT_EXTENT_VIEW,
075        Arrays.asList(
076            DefaultSnowballStemmer.getDescription("English"),
077            PosTaggerAnnotator.getDescription(),
078            ParserAnnotator.getDescription()),
079        infos);
080  }
081}
