109      @Override
110      public void initialize(UimaContext context) throws ResourceInitializationException {
111        super.initialize(context);
112    
113        SimpleFeatureExtractor defaultTokenExtractorSet = new MatchingAnnotationExtractor(
114            Token.class,
115            new SpannedTextExtractor(),
116            new StemExtractor(),
117            new POSExtractor());
118    
119        SimpleFeatureExtractor defaultConstituentExtractorSet = new CombinedExtractor(
120            new NodeTypeExtractor(),
121            // new TypePathExtractor(TreebankNode.class, "nodeTags"),
122            new HeadWordExtractor(defaultTokenExtractorSet),
123            new FirstInstanceExtractor(Token.class, defaultTokenExtractorSet),
124            new LastInstanceExtractor(Token.class, defaultTokenExtractorSet));
125    
126        this.predicateExtractor = new NamingExtractor(
127            "Predicate",
128            defaultTokenExtractorSet,
129            new MatchingAnnotationExtractor(TreebankNode.class, new SubCategorizationExtractor()));
130    
131        this.constituentExtractor = new NamingExtractor("Constituent", new MatchingAnnotationExtractor(
132            TreebankNode.class,
133            defaultConstituentExtractorSet,
134            new ParentExtractor(defaultConstituentExtractorSet),
135            new SiblingExtractor(-1, defaultConstituentExtractorSet),
136            new SiblingExtractor(1, defaultConstituentExtractorSet),
137            new NamedEntityExtractor()));
138    
139        this.predicateAndConstituentExtractor = new NamingAnnotationPairFeatureExtractor(
140            "PredicateAndConstituent",
141            new MatchingAnnotationPairExtractor(
142                TreebankNode.class,
143                TreebankNode.class,
144                new SyntacticPathExtractor(new NodeTypeExtractor()),
145                new RelativePositionExtractor()));
146    
147      }

149      @Override
150      public void process(JCas jCas) throws AnalysisEngineProcessException {
151        /*
152         * Iterate over sentences in document
153         */
154        Collection<Sentence> sentences = JCasUtil.select(jCas, Sentence.class);
155    
156        nSentences = 0;
157        nPredicates = 0;
158        nArguments = 0;
159    
160        for (Sentence sentence : sentences) {
161          processSentence(jCas, sentence);
162        }
163    
164        logger.info(String.format(
165            "processed %d sentences, %d predicates, ~%d arguments per predicate",
166            nSentences,
167            nPredicates,
168            nPredicates == 0 ? 0 : nArguments / nPredicates));
169      }
170    
171      void processSentence(JCas jCas, Sentence sentence) throws AnalysisEngineProcessException {
172        nSentences += 1;
173    
174        if (sentence.getCoveredText().length() > 40)
175          logger.fine(String.format(
176              "process sentence \"%s ...\"",
177              sentence.getCoveredText().substring(0, 39)));
178        else
179          logger.fine(String.format("process sentence \"%s\"", sentence.getCoveredText()));
180    
181        /*
182         * Pre-compute sentence level data: sentenceConstituents: list of all constituents in sentence
183         */
184        TopTreebankNode top = AnnotationUtil.selectFirstMatching(jCas, TopTreebankNode.class, sentence);
185        if (top == null) {
186          throw CleartkExtractorException.noAnnotationInWindow(TopTreebankNode.class, sentence);
187        }
188        List<TreebankNode> constituents = new ArrayList<TreebankNode>(200);
189        collectConstituents(top, constituents);
190    
191        /*
192         * Compute constituent features for all constituents in sentence
193         */
194        Map<TreebankNode, List<Feature>> constituentFeatures = new HashMap<TreebankNode, List<Feature>>();
195        for (TreebankNode c : constituents) {
196          List<Feature> features = constituentExtractor.extract(jCas, c);
197          constituentFeatures.put(c, features);
198        }
199    
200        /*
201         * Iterate over predicates in sentence
202         */
203        List<Predicate> predicates = JCasUtil.selectCovered(jCas, Predicate.class, sentence);
204        for (Predicate predicate : predicates) {
205          processPredicate(jCas, predicate, constituentFeatures);
206        }
207      }
208    
209      public void processPredicate(
210          JCas jCas,
211          Predicate predicate,
212          Map<TreebankNode, List<Feature>> sentenceConstituentFeatures)
213          throws AnalysisEngineProcessException {
214        nPredicates += 1;
215    
216        /*
217         * Compute predicate features
218         */
219        List<Feature> predicateFeatures = new ArrayList<Feature>(12);
220        predicateFeatures.addAll(predicateExtractor.extract(jCas, predicate));
221    
222        List<SemanticArgument> arguments = UIMAUtil.toList(
223            predicate.getArguments(),
224            SemanticArgument.class);
225    
226        /*
227         * Iterate over arguments
228         */
229        for (SemanticArgument arg : arguments) {
230          if (!(arg.getAnnotation() instanceof TreebankNode)) {
231            logger.warning(String.format(
232                "skipping argument of \"%s\", because it doesn't align with the parse tree",
233                predicate.getCoveredText()));
234            continue;
235          }
236    
237          nArguments += 1;
238    
239          TreebankNode constituent = (TreebankNode) arg.getAnnotation();
240    
241          Instance<String> instance = new Instance<String>();
242    
243          /*
244           * Compute predicate-constituent features
245           */
246          instance.addAll(predicateAndConstituentExtractor.extract(jCas, predicate, constituent));
247    
248          /*
249           * Add constituent features
250           */
251          instance.addAll(sentenceConstituentFeatures.get(constituent));
252    
253          /*
254           * Add predicate features
255           */
256          instance.addAll(predicateFeatures);
257    
258          if (isTraining()) {
259            instance.setOutcome(arg.getLabel());
260            this.dataWriter.write(instance);
261          } else {
262            arg.setLabel(this.classifier.classify(instance.getFeatures()));
263          }
264    
265        }
266      }
267    
268      /**
269       * Recursively build a list of constituents under a TreebankNode.
270       * 
271       * @param top
272       *          the root of the parse tree to operate on; <b>top</b> itself will also be added, unless
273       *          it is of type TopTrebankNode
274       * @param constituents
275       *          list of nodes to add to
276       */
277      protected void collectConstituents(TreebankNode top, List<TreebankNode> constituents) {
278        if (!(top instanceof TopTreebankNode))
279          constituents.add(top);
280    
281        if (top.getChildren() == null)
282          return;
283    
284        int numberOfChildren = top.getChildren().size();
285        for (int i = 0; i < numberOfChildren; i++) {
286          collectConstituents(top.getChildren(i), constituents);
287        }
288      }
