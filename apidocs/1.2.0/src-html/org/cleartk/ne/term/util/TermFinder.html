024    package org.cleartk.ne.term.util;
025    
026    import java.util.List;
027    
028    import org.cleartk.token.tokenizer.Token;
029    
030    /**
031     * <br>
032     * Copyright (c) 2007-2008, Regents of the University of Colorado <br>
033     * All rights reserved.
034     * 
035     * <p>
036     * 
037     * This interface defines a term finder - a class which finds terms provided in one or more term
038     * lists in text.
039     * 
040     * @author Philip
041     * 
042     */
043    public interface TermFinder {
044      /**
045       * 
046       * @param tokens
047       *          a list of tokens corresponding to the text to look for terms in.
048       * 
049       * @return all of the terms that were matched to spans of text
050       */
051      public List<TermMatch> getMatches(List<Token> tokens);
052    
053      /**
054       * A term finder should typically be able to handle more than one term list. Furthermore, a term
055       * list makes no guarantees that the terms are unique with respect to the term texts either within
056       * a term list or across multiple term lists.
057       */
058      public void addTermList(TermList termList);
059    }
