037    /**
038     * A factory class that simplifies the creation of descriptors for {@link CleartkAnnotator} and
039     * {@link CleartkSequenceAnnotator} classes. The factory methods here wrap common patterns of calls
040     * to {@link AnalysisEngineFactory#createPrimitiveDescription} and
041     * {@link ConfigurationParameterFactory#addConfigurationParameter} for setting classifier classes,
042     * output directories, etc. that are necessary when creating a CleartkAnnotator or
043     * CleartkSequenceAnnotator.
044     * 
045     * <br>
046     * Copyright (c) 2009, Regents of the University of Colorado <br>
047     * All rights reserved.
048     * 
049     * @author Steven Bethard
050     * @deprecated Use {@link AnalysisEngineFactory} directly.
051     */
052    @Deprecated
053    public class CleartkAnnotatorDescriptionFactory {
054    
055      /**
056       * Create an {@link AnalysisEngineDescription} for using a {@link CleartkSequenceAnnotator} to
057       * write training data for a {@link ViterbiClassifier}. Note that data is written with a
058       * {@link DataWriterFactory}, not a {@link SequenceDataWriterFactory} - A ViterbiClassifier wraps
059       * a non-sequence classifier to work as a sequence classifier.
060       * 
061       * @param <OUTCOME_TYPE>
062       *          The outcome type of the classifier (must be the same for both the
063       *          CleartkSequenceAnnotator and the DataWriterFactory).
064       * @param annotatorClass
065       *          The main annotator for the AnalysisEngineDescription.
066       * @param typeSystemDescription
067       *          The type system used by the annotator.
068       * @param delegatedDataWriterFactoryClass
069       *          The non-sequence DataWriterFactory that will be used to write the training data (and
070       *          indicate the type of classifier that will be trained from this data).
071       * @param outputDir
072       *          The directory where the training data should be written.
073       * @return An AnalysisEngineDescription for the CleartkSequenceAnnotator.
074       * @deprecated Use {@link AnalysisEngineFactory} directly.
075       */
076      @Deprecated
077      public static <OUTCOME_TYPE> AnalysisEngineDescription createViterbiAnnotator(
078          Class<? extends CleartkSequenceAnnotator<OUTCOME_TYPE>> annotatorClass,
079          TypeSystemDescription typeSystemDescription,
080          Class<? extends DataWriterFactory<OUTCOME_TYPE>> delegatedDataWriterFactoryClass,
081          String outputDir) throws ResourceInitializationException {
082    
083        return AnalysisEngineFactory.createPrimitiveDescription(
084            annotatorClass,
085            typeSystemDescription,
086            CleartkSequenceAnnotator.PARAM_DATA_WRITER_FACTORY_CLASS_NAME,
087            ViterbiDataWriterFactory.class.getName(),
088            DirectoryDataWriterFactory.PARAM_OUTPUT_DIRECTORY,
089            outputDir,
090            ViterbiDataWriterFactory.PARAM_DELEGATED_DATA_WRITER_FACTORY_CLASS,
091            delegatedDataWriterFactoryClass.getName(),
092            ViterbiDataWriterFactory.PARAM_OUTCOME_FEATURE_EXTRACTOR_NAMES,
093            new String[] { DefaultOutcomeFeatureExtractor.class.getName() });
094      }
095    
096      /**
097       * Create an {@link AnalysisEngineDescription} for using a {@link CleartkAnnotator} to create new
098       * annotations based on the predictions of a classifier.
099       * 
100       * @param cleartkAnnotatorClass
101       *          The main annotator for the AnalysisEngineDescription.
102       * @param typeSystemDescription
103       *          The type system used by the annotator.
104       * @param classifierJar
105       *          The jar file containing the classifier that will make predictions.
106       * @return An AnalysisEngineDescription for the CleartkAnnotator.
107       * @deprecated Use {@link AnalysisEngineFactory} directly.
108       */
109      @Deprecated
110      public static AnalysisEngineDescription createCleartkAnnotator(
111          Class<? extends CleartkAnnotator<?>> cleartkAnnotatorClass,
112          TypeSystemDescription typeSystemDescription,
113          String classifierJar) throws ResourceInitializationException {
114    
115        AnalysisEngineDescription aed = AnalysisEngineFactory.createPrimitiveDescription(
116            cleartkAnnotatorClass,
117            typeSystemDescription);
118        ConfigurationParameterFactory.addConfigurationParameter(
119            aed,
120            GenericJarClassifierFactory.PARAM_CLASSIFIER_JAR_PATH,
121            classifierJar);
122        return aed;
123      }
124

138       *          type of classifier that will be trained from this data).
139       * @param outputDir
140       *          The directory where the training data should be written.
141       * @return An AnalysisEngineDescription for the CleartkAnnotator.
142       * @deprecated Use {@link AnalysisEngineFactory} directly.

157      }
