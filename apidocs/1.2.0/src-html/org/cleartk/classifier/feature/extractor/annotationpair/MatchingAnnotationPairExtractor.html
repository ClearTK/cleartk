032    import org.cleartk.util.AnnotationUtil;

040     */
041    public class MatchingAnnotationPairExtractor implements AnnotationPairFeatureExtractor {
042    
043      public MatchingAnnotationPairExtractor(
044          Class<? extends Annotation> leftType,
045          Class<? extends Annotation> rightType,
046          AnnotationPairFeatureExtractor subExtractor) {
047        this.leftType = leftType;
048        this.rightType = rightType;
049        this.subExtractor = subExtractor;
050      }
051    
052      public MatchingAnnotationPairExtractor(
053          Class<? extends Annotation> leftType,
054          Class<? extends Annotation> rightType,
055          AnnotationPairFeatureExtractor... subExtractors) {
056        this(leftType, rightType, new CombinedAnnotationPairFeatureExtractor(subExtractors));
057      }
058    
059      public List<Feature> extract(JCas view, Annotation leftAnnotation, Annotation rightAnnotation)
060          throws CleartkExtractorException {
061    
062        Annotation leftMatchingAnnotation;
063        Annotation rightMatchingAnnotation;
064    
065        if (!leftType.isInstance(leftAnnotation)) {
066          leftMatchingAnnotation = AnnotationUtil.selectFirstMatching(view, leftType, leftAnnotation);
067          if (leftMatchingAnnotation == null) {
068            throw CleartkExtractorException.noAnnotationMatchingWindow(leftType, leftAnnotation);
069          }
070        } else {
071          leftMatchingAnnotation = leftType.cast(leftAnnotation);
072        }
073    
074        if (!rightType.isInstance(rightAnnotation)) {
075          rightMatchingAnnotation = AnnotationUtil.selectFirstMatching(view, rightType, rightAnnotation);
076          if (rightMatchingAnnotation == null) {
077            throw CleartkExtractorException.noAnnotationMatchingWindow(rightType, rightAnnotation);
078          }
079        } else {
080          rightMatchingAnnotation = rightType.cast(rightAnnotation);
081        }
082    
083        return subExtractor.extract(view, leftMatchingAnnotation, rightMatchingAnnotation);
084      }
085    
086      private Class<? extends Annotation> leftType;
087    
088      private Class<? extends Annotation> rightType;
089    
090      private AnnotationPairFeatureExtractor subExtractor;
091    
092    }
