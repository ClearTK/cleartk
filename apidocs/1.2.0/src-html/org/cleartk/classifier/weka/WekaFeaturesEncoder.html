040    public class WekaFeaturesEncoder implements FeaturesEncoder<Iterable<Feature>> {
041    
042      private static final long serialVersionUID = 1L;
043    
044      private ArrayList<Attribute> attributes;
045    
046      private Map<String, Attribute> attributeMap;
047    
048      public WekaFeaturesEncoder() {
049        attributes = new ArrayList<Attribute>();
050        attributeMap = new HashMap<String, Attribute>();
051      }
052    
053      public Iterable<Feature> encodeAll(Iterable<Feature> features) {
054        for (Feature feature : features) {
055          featureToAttribute(feature);
056        }
057        return features;
058      }
059    
060      /**
061       * @param feature
062       * @return
063       */
064      private Attribute featureToAttribute(Feature feature) {
065        String name = feature.getName();
066        Attribute attribute = attributeMap.get(name);
067        if (attribute == null) {
068          attribute = featureToAttribute(feature, attributes.size());
069          attributes.add(attribute);
070          attributeMap.put(name, attribute);
071        }
072        return attribute;
073      }
074    
075      public static Attribute featureToAttribute(Feature feature, int attributeIndex) {
076        String name = Utils.quote(feature.getName());
077        Object value = feature.getValue();
078        Attribute attribute;
079        // if value is a number then create a numeric attribute
080        if (value instanceof Number) {
081          attribute = new Attribute(name);
082        }// if value is a boolean then create a numeric attribute
083        else if (value instanceof Boolean) {
084          attribute = new Attribute(name);
085        }
086        // if value is an Enum thene create a nominal attribute
087        else if (value instanceof Enum) {
088          Object[] enumConstants = value.getClass().getEnumConstants();
089          ArrayList<String> attributeValues = new ArrayList<String>(enumConstants.length);
090          for (Object enumConstant : enumConstants) {
091            attributeValues.add(enumConstant.toString());
092          }
093          attribute = new Attribute(name, attributeValues);
094        }
095        // if value is not a number, boolean, or enum, then we will create a
096        // string attribute
097        else {
098          attribute = new Attribute(name, (ArrayList<String>) null);
099        }
100        return attribute;
101      }
102    
103      public void finalizeFeatureSet(File outputDirectory) {
104      }
105    
106      public ArrayList<Attribute> getWekaAttributes() {
107        return attributes;
108      }
109    
110      public Map<String, Attribute> getWekaAttributeMap() {
111        return attributeMap;
112      }
113    
114    }
