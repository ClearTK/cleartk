024    package org.cleartk.classifier.opennlp;
025    
026    import java.io.DataInputStream;
027    import java.io.File;
028    import java.io.IOException;
029    import java.util.List;
030    import java.util.jar.JarInputStream;
031    import java.util.jar.JarOutputStream;
032    import java.util.zip.GZIPInputStream;
033    
034    import opennlp.maxent.io.BinaryGISModelReader;
035    import opennlp.model.MaxentModel;
036    
037    import org.cleartk.classifier.encoder.features.NameNumber;
038    import org.cleartk.classifier.encoder.features.NameNumberFeaturesEncoder;
039    import org.cleartk.classifier.jar.ClassifierBuilder_ImplBase;
040    import org.cleartk.classifier.jar.JarStreams;
041    
042    /**
043     * <br>
044     * Copyright (c) 2007-2011, Regents of the University of Colorado <br>
045     * All rights reserved.
046     * 
047     * @author Philip Ogren
048     * @author Steven Bethard
049     */
050    public abstract class MaxentClassifierBuilder_ImplBase<CLASSIFIER_TYPE extends MaxentClassifier_ImplBase<OUTCOME_TYPE>, OUTCOME_TYPE>
051        extends ClassifierBuilder_ImplBase<CLASSIFIER_TYPE, List<NameNumber>, OUTCOME_TYPE, String> {
052    
053      private static final String MODEL_NAME = "model.maxent";
054    
055      public File getTrainingDataFile(File dir) {
056        return new File(dir, "training-data.maxent");
057      }
058    
059      public void trainClassifier(File dir, String... args) throws Exception {
060        String[] maxentArgs = new String[args.length + 1];
061        maxentArgs[0] = getTrainingDataFile(dir).getPath();
062        System.arraycopy(args, 0, maxentArgs, 1, args.length);
063        opennlp.model.RealValueFileEventStream.main(maxentArgs);
064      }
065    
066      @Override
067      protected void packageClassifier(File dir, JarOutputStream modelStream) throws IOException {
068        super.packageClassifier(dir, modelStream);
069        JarStreams.putNextJarEntry(modelStream, MODEL_NAME, new File(dir, "training-data.maxent.bin.gz"));
070        File featureLookup = new File(dir, NameNumberFeaturesEncoder.LOOKUP_FILE_NAME);
071        if (featureLookup.exists()) {
072          JarStreams.putNextJarEntry(modelStream, "name-lookup.txt", featureLookup);
073        }
074      }
075    
076      protected MaxentModel model;
077    
078      @Override
079      protected void unpackageClassifier(JarInputStream modelStream) throws IOException {
080        super.unpackageClassifier(modelStream);
081        JarStreams.getNextJarEntry(modelStream, MODEL_NAME);
082        this.model = new BinaryGISModelReader(new DataInputStream(new GZIPInputStream(modelStream)))
083            .getModel();
084      }
085    
086    }
