055public class ParserAnnotator<TOKEN_TYPE extends Annotation, SENTENCE_TYPE extends Annotation, TOP_NODE_TYPE extends Annotation>
056    extends ParserWrapper_ImplBase<TOKEN_TYPE, SENTENCE_TYPE, Tree<String>, TOP_NODE_TYPE> {
057
058  public static final String PARAM_PARSER_MODEL_PATH = ConfigurationParameterFactory.createConfigurationParameterName(
059      ParserAnnotator.class,
060      "parserModelPath");
061
062  @ConfigurationParameter
063  private String parserModelPath;
064
065  protected CoarseToFineMaxRuleParser parser;
066
067  private int parseFailureCount = 0;
068
069  private int sentenceCount = 0;
070
071  @Override
072  public void initialize(UimaContext context) throws ResourceInitializationException {
073    super.initialize(context);
074    System.out.print("initializing Berkeley Parser: " + parserModelPath + " ... ");
075
076    long start = System.nanoTime();
077
078    InputStream modelInputStream;
079    try {
080      modelInputStream = IOUtil.getInputStream(ParserAnnotator.class, parserModelPath);
081
082      GZIPInputStream gzis = new GZIPInputStream(modelInputStream); // Compressed
083      ObjectInputStream in = new ObjectInputStream(gzis); // Load objects
084      ParserData parserData = (ParserData) in.readObject(); // Read the mix of grammars
085
086      Grammar grammar = parserData.getGrammar();
087      Lexicon lexicon = parserData.getLexicon();
088      Numberer.setNumberers(parserData.getNumbs());
089      parser = new CoarseToFineMaxRuleParser(
090          grammar,
091          lexicon,
092          1.0,
093          -1,
094          false,
095          false,
096          false,
097          false,
098          false,
099          true,
100          true);
101      long stop = System.nanoTime();
102      float seconds = (float) (stop - start) / 1000000000;
103      System.out.println("done.   Loaded in: " + seconds + " seconds");
104    } catch (IOException e) {
105      throw new ResourceInitializationException(e);
106    } catch (ClassNotFoundException e) {
107      throw new ResourceInitializationException(e);
108    }
109
110  }
111
112  @Override
113  public void process(JCas jCas) throws AnalysisEngineProcessException {
114
115    List<SENTENCE_TYPE> sentenceList = inputTypesHelper.getSentences(jCas);
116
117    for (SENTENCE_TYPE sentence : sentenceList) {
118      sentenceCount++;
119      List<TOKEN_TYPE> tokens = inputTypesHelper.getTokens(jCas, sentence);
120      List<String> words = new ArrayList<String>();
121      List<String> tags = new ArrayList<String>();
122
123      for (TOKEN_TYPE token : tokens) {
124        words.add(token.getCoveredText());
125        String tag = inputTypesHelper.getPosTag(token);
126        tags.add(tag);
127      }
128
129      Tree<String> tree = parser.getBestConstrainedParse(words, tags, null);
130      if (tree.isLeaf()) {
131        System.out.println("words: " + words.size() + "  " + words);
132        System.out.println("tags: " + tags.size() + "  " + tags);
133        System.out.println("unable to parse sentence: " + sentence.getCoveredText());
134        parseFailureCount++;
135      } else {
136        outputTypesHelper.addParse(jCas, tree, sentence, tokens);
137      }
138    }
139  }
140
141  @Override
142  public void collectionProcessComplete() throws AnalysisEngineProcessException {
143    System.out.println("total number of sentences that were not parsed was: " + parseFailureCount
144        + " out of " + sentenceCount);
145  }
