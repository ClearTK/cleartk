067/**
068 * <p>
069 * This class can be used to train and test the named entity chunker. It can be used to either
070 * perform 2-fold cross-validation, or training and testing on a holdout test set, or just to
071 * evaluate/test a pre-trained model.
072 * 
073 * <br>
074 * Copyright (c) 2012, Regents of the University of Colorado <br>
075 * All rights reserved.
076 * 
077 * @author Himanshu Gahlot
078 * @author Steven Bethard
079 */
080public class EvaluateNamedEntityChunker extends
081    Evaluation_ImplBase<File, AnnotationStatistics<String>> {
082
083  public static class Options extends Options_ImplBase {
084    @Option(
085        name = "--train-dir",
086        usage = "Specify the directory containing the training documents.  This is used for cross-validation and for training in a holdout set evaluator. "
087            + "When we run this example we point to a directory containing training data from the MASC-1.0.3 corpus - i.e. a directory called 'MASC-1.0.3/data/written'")
088    public File trainDirectory = new File("src/main/resources/data/MASC-1.0.3/data/written");
089
090    @Option(
091        name = "--models-dir",
092        usage = "specify the directory in which to write out the trained model files")
093    public File modelsDirectory = new File("target/chunking/ne-model");
094  }
095
096  public static void main(String[] args) throws Exception {
097    Options options = new Options();
098    options.parseOptions(args);
099
100    // find training files
101    List<File> trainFiles = new ArrayList<File>(FileUtils.listFiles(
102        options.trainDirectory,
103        new MASCTextFileFilter(),
104        FileFilterUtils.falseFileFilter()));
105
106    // run cross validation
107    EvaluateNamedEntityChunker evaluator = new EvaluateNamedEntityChunker(options.modelsDirectory);
108    List<AnnotationStatistics<String>> foldStats = evaluator.crossValidation(trainFiles, 2);
109    AnnotationStatistics<String> crossValidationStats = AnnotationStatistics.addAll(foldStats);
110
111    System.err.println("Cross Validation Results:");
112    System.err.print(crossValidationStats);
113    System.err.println();
114    System.err.println(crossValidationStats.confusions());
115    System.err.println();
116
117    // train and save a model using all the data
118    evaluator.trainAndTest(trainFiles, Collections.<File> emptyList());
119  }
120
121  public EvaluateNamedEntityChunker(File baseDirectory) {
122    super(baseDirectory);
123  }
124
125  @Override
126  protected CollectionReader getCollectionReader(List<File> files) throws Exception {
127    return CollectionReaderFactory.createCollectionReader(UriCollectionReader.getDescriptionFromFiles(files));
128  }
129
130  @Override
131  public void train(CollectionReader collectionReader, File outputDirectory) throws Exception {
132    // assemble the training pipeline
133    AggregateBuilder aggregate = new AggregateBuilder();
134
135    // an annotator that loads the text from the training file URIs
136    aggregate.add(UriToDocumentTextAnnotator.getDescription());
137
138    // an annotator that parses and loads MASC named entity annotations (and tokens)
139    aggregate.add(MASCGoldAnnotator.getDescription());
140
141    // an annotator that adds part-of-speech tags
142    aggregate.add(PosTaggerAnnotator.getDescription());
143
144    // our NamedEntityChunker annotator, configured to write Mallet CRF training data
145    aggregate.add(AnalysisEngineFactory.createPrimitiveDescription(
146        NamedEntityChunker.class,
147        CleartkSequenceAnnotator.PARAM_IS_TRAINING,
148        true,
149        DirectoryDataWriterFactory.PARAM_OUTPUT_DIRECTORY,
150        outputDirectory,
151        DefaultSequenceDataWriterFactory.PARAM_DATA_WRITER_CLASS_NAME,
152        MalletCRFStringOutcomeDataWriter.class));
153
154    // run the pipeline over the training corpus
155    SimplePipeline.runPipeline(collectionReader, aggregate.createAggregateDescription());
156
157    // quiet Mallet down a bit (but still leave likelihoods so you can see progress)
158    Logger malletLogger = Logger.getLogger("cc.mallet");
159    malletLogger.setLevel(Level.WARNING);
160    Logger likelihoodLogger = Logger.getLogger("cc.mallet.fst.CRFOptimizableByLabelLikelihood");
161    likelihoodLogger.setLevel(Level.INFO);
162
163    // train a Mallet CRF model on the training data
164    Train.main(outputDirectory);
165
166  }
167
168  @Override
169  protected AnnotationStatistics<String> test(CollectionReader collectionReader, File modelDirectory)
170      throws Exception {
171
172    final String defaultViewName = CAS.NAME_DEFAULT_SOFA;
173    final String goldViewName = "GoldView";
174
175    // define the pipeline
176    AggregateBuilder aggregate = new AggregateBuilder();
177
178    // Annotators processing the gold view:
179    // * create the gold view
180    // * load the text
181    // * load the MASC annotations
182    aggregate.add(AnalysisEngineFactory.createPrimitiveDescription(
183        ViewCreatorAnnotator.class,
184        ViewCreatorAnnotator.PARAM_VIEW_NAME,
185        goldViewName));
186    aggregate.add(UriToDocumentTextAnnotator.getDescription(), defaultViewName, goldViewName);
187    aggregate.add(MASCGoldAnnotator.getDescription(), defaultViewName, goldViewName);
188
189    // Annotators processing the default (system) view:
190    // * load the text
191    // * parse sentences, tokens, part-of-speech tags
192    // * run the named entity chunker
193    aggregate.add(UriToDocumentTextAnnotator.getDescription());
194    aggregate.add(SentenceAnnotator.getDescription());
195    aggregate.add(TokenAnnotator.getDescription());
196    aggregate.add(PosTaggerAnnotator.getDescription());
197    aggregate.add(AnalysisEngineFactory.createPrimitiveDescription(
198        NamedEntityChunker.class,
199        CleartkSequenceAnnotator.PARAM_IS_TRAINING,
200        false,
201        GenericJarClassifierFactory.PARAM_CLASSIFIER_JAR_PATH,
202        new File(modelDirectory, "model.jar")));
203
204    // prepare the evaluation statistics
205    AnnotationStatistics<String> stats = new AnnotationStatistics<String>();
206    Function<NamedEntityMention, ?> getSpan = AnnotationStatistics.annotationToSpan();
207    Function<NamedEntityMention, String> getCategory = AnnotationStatistics.annotationToFeatureValue("mentionType");
208
209    // iterate over each JCas to be evaluated
210    for (JCas jCas : new JCasIterable(collectionReader, aggregate.createAggregate())) {
211      JCas goldView = jCas.getView(goldViewName);
212      JCas systemView = jCas.getView(defaultViewName);
213
214      // extract the named entity mentions from both gold and system views
215      Collection<NamedEntityMention> goldMentions, systemMentions;
216      goldMentions = JCasUtil.select(goldView, NamedEntityMention.class);
217      systemMentions = JCasUtil.select(systemView, NamedEntityMention.class);
218
219      // compare the system mentions to the gold mentions
220      stats.add(goldMentions, systemMentions, getSpan, getCategory);
221    }
222
223    return stats;
224  }
