025package org.cleartk.token.breakit;

027import java.lang.reflect.Constructor;
028import java.text.BreakIterator;
029import java.util.Locale;
030
031import org.apache.uima.UimaContext;
032import org.apache.uima.analysis_engine.AnalysisEngineProcessException;
033import org.apache.uima.jcas.JCas;
034import org.apache.uima.jcas.tcas.Annotation;
035import org.apache.uima.resource.ResourceInitializationException;
036import org.uimafit.component.JCasAnnotator_ImplBase;
037import org.uimafit.descriptor.ConfigurationParameter;
038import org.uimafit.factory.ConfigurationParameterFactory;
039import org.uimafit.factory.initializable.InitializableFactory;
040

049
050public class BreakIteratorAnnotator extends JCasAnnotator_ImplBase {
051
052  public static final String PARAM_LOCALE = ConfigurationParameterFactory
053      .createConfigurationParameterName(BreakIteratorAnnotator.class, "locale");
054
055  @ConfigurationParameter(description = "provides the name of the locale to be used to instantiate the break iterator")
056  private Locale locale;
057
058  public static enum BreakIteratorType {
059    WORD, SENTENCE
060  }
061
062  public static final String PARAM_BREAK_ITERATOR_TYPE = ConfigurationParameterFactory
063      .createConfigurationParameterName(BreakIteratorAnnotator.class, "breakIteratorType");
064
065  @ConfigurationParameter(description = "provides the type of the locale to be used to instantiate the break iterator.  Should be one of  'WORD' or 'SENTENCE'", defaultValue = "SENTENCE")
066  private BreakIteratorType breakIteratorType;
067
068  public static final String PARAM_ANNOTATION_TYPE_NAME = ConfigurationParameterFactory
069      .createConfigurationParameterName(BreakIteratorAnnotator.class, "annotationTypeName");
070
071  @ConfigurationParameter(description = "class type of the annotations that are created by this annotator.")
072  private String annotationTypeName;
073
074  private Class<? extends Annotation> annotationClass;
075
076  private Constructor<? extends Annotation> annotationConstructor;
077
078  private BreakIterator breakIterator;
079
080  @Override
081  public void initialize(UimaContext context) throws ResourceInitializationException {
082    super.initialize(context);
083
084    try {
085      annotationClass = InitializableFactory.getClass(annotationTypeName, Annotation.class);
086      annotationConstructor = annotationClass.getConstructor(new Class[] {
087          JCas.class,
088          Integer.TYPE,
089          Integer.TYPE });
090    } catch (Exception e) {
091      throw new ResourceInitializationException(e);
092    }
093
094    if (locale == null) {
095      locale = Locale.getDefault();
096    }
097    if (breakIteratorType == BreakIteratorType.WORD) {
098      breakIterator = BreakIterator.getWordInstance(locale);
099    } else {
100      breakIterator = BreakIterator.getSentenceInstance(locale);
101    }
102
103  }
104
105  @Override
106  public void process(JCas jCas) throws AnalysisEngineProcessException {
107    String text = jCas.getDocumentText();
108    breakIterator.setText(text);
109
110    int index = breakIterator.first();
111    int endIndex;
112    while ((endIndex = breakIterator.next()) != BreakIterator.DONE) {
113      String annotationText = text.substring(index, endIndex);
114      if (!annotationText.trim().equals("")) {
115        try {
116          annotationConstructor.newInstance(jCas, index, endIndex).addToIndexes();
117        } catch (Exception e) {
118          throw new AnalysisEngineProcessException(e);
119        }
120      }
121      index = endIndex;
122    }
123  }
124
125}
